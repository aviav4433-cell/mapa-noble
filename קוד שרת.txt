function setupDatabase() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  Object.keys(TABLES).forEach(function (name) {
    var sh = ss.getSheetByName(name);
    if (!sh) {
      sh = ss.insertSheet(name);
      sh.getRange(1, 1, 1, TABLES[name].length).setValues([TABLES[name]])
        .setFontWeight('bold').setBackground('#1F4E78').setFontColor('#FFFFFF');
      sh.setFrozenRows(1);
    } else {
      // מיגרציה: הוספת עמודות חדשות שנוספו בגרסה — לסוף בלבד
      var cols = TABLES[name];
      var existing = sh.getRange(1, 1, 1, cols.length).getValues()[0];
      for (var ci = 0; ci < cols.length; ci++) {
        if (existing[ci] !== cols[ci]) {
          sh.getRange(1, ci + 1, 1, 1).setValues([[cols[ci]]])
            .setFontWeight('bold').setBackground('#1F4E78').setFontColor('#FFFFFF');
        }
      }
    }
  });
  var def = ss.getSheetByName('Sheet1') || ss.getSheetByName('גיליון1');
  if (def && ss.getSheets().length > 1) ss.deleteSheet(def);
  // אצווה D: גליון מילון עברי — טאב ראשון שמסביר כל גליון (אין לשנות שמות באנגלית!)

// ---------- Data access ----------
function readTable(name) {
  var sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sh || sh.getLastRow() < 2) return [];
  var cols = TABLES[name];
  var vals = sh.getRange(2, 1, sh.getLastRow() - 1, cols.length).getValues();
  return vals.filter(function (r) { return r[0] !== ''; }).map(function (r) {
    var o = {};
    cols.forEach(function (c, i) {
      var v = r[i];
      if (v instanceof Date) v = Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      o[c] = v;
    });
    return o;
  });
}
function writeTable(name, rows) {
  var sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  var cols = TABLES[name];
  if (sh.getLastRow() > 1) sh.getRange(2, 1, sh.getLastRow() - 1, cols.length).clearContent();
  if (rows.length) {
    var vals = rows.map(function (o) { return cols.map(function (c) { return o[c] !== undefined ? o[c] : ''; }); });
    sh.getRange(2, 1, vals.length, cols.length).setValues(vals);
  }
}

// F1: טעינה עצלה — כל טבלה נקראת מהגליון רק בגישה הראשונה אליה (getter + מטמון)
// ה-getters הם enumerable כדי ש-JSON.stringify ב-getAll ימשיך להחזיר את כל הטבלאות,
// וקיים setter כי חלק מהקוד מבצע השמה ישירה (למשל db.orderLines = db.orderLines.filter(...))
// G1: שדות טקסט ארוכים — נחתכים בתשובת getAll ונשלפים במלואם רק לפי דרישה.
// הנתונים בגליון עצמו לעולם אינם נחתכים — הקיצוץ הוא רק בעותק שנשלח לממשק.
var LONG_TEXT_PREVIEW = 300;
var LONG_TEXT_FIELDS = { notes: ['text'], tasks: ['details'], meetings: ['protocol'], task_comments: ['text'] };

// G1: בונה את מטען getAll — עותק של הטבלאות שבו שדות ארוכים קוצצו וסומנו ב-<field>_trunc
function getAllPayload(db) {
  var out = {};
  Object.keys(TMAP).forEach(function (t) {
    var key = TMAP[t];
    var rows = db[key];
    var fields = LONG_TEXT_FIELDS[t];
    if (!fields) { out[key] = rows; return; }
    out[key] = rows.map(function (r) {
      var needs = fields.some(function (f) { return typeof r[f] === 'string' && r[f].length > LONG_TEXT_PREVIEW; });
      if (!needs) return r;
      var c = {};
      for (var k in r) c[k] = r[k];
      fields.forEach(function (f) {
        if (typeof c[f] === 'string' && c[f].length > LONG_TEXT_PREVIEW) {
          c[f] = c[f].substring(0, LONG_TEXT_PREVIEW);
          c[f + '_trunc'] = 1;
        }
      });
      return c;
    });
  });
  return out;
}

function loadDb() {
  var db = {};
  var cache = {};
  Object.keys(TMAP).forEach(function (t) {
    var key = TMAP[t];
    Object.defineProperty(db, key, {
      enumerable: true,
      configurable: true,
      get: function () {
        if (!(key in cache)) cache[key] = readTable(t);
        return cache[key];
      },
      set: function (v) { cache[key] = v; }
    });
  });
  return db;
}
// F1: הוספת שורה בודדת לגליון (במקום שכתוב כל הטבלה) — אותה סריאליזציה כמו writeTable
function appendRowToTable(name, o) {
  var sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  var cols = TABLES[name];
  sh.appendRow(cols.map(function (c) { return o[c] !== undefined ? o[c] : ''; }));
}
function genId(prefix) {
  return prefix + '-' + new Date().getTime().toString(36).toUpperCase() + Math.floor(Math.random() * 900 + 100);
}
function today() { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd'); }
function nowTs() { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss'); }

// מונה מסמכים רץ (חשבוניות מ-1001, קבלות מ-5001)
function nextNumber(db, key, start) {
  var c = db.counters.find(function (x) { return x.key === key; });
  if (!c) { c = { key: key, value: start }; db.counters.push(c); }
  else { c.value = Number(c.value) + 1; }
  writeTable('counters', db.counters);
  return Number(c.value);
}
// F1: כתיבת שורה בודדת ליומן (appendRow) — במקום שכתוב כל גליון היומן בכל פעולה.

// ---------- Security helpers ----------
function sha256Hex(str) {
  var bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, str, Utilities.Charset.UTF_8);
  return bytes.map(function (b) { return ('0' + (b & 0xff).toString(16)).slice(-2); }).join('');
}

function sha256WithSalt(password, salt) {
  return sha256Hex(String(salt || '') + ':' + String(password || ''));
}

// B7: "מנהל ראשי" — עוגן אמון עצמאי שלא תלוי בשדה role בגיליון (מונע נעילה מוחלטת אם השדה נפגע/נמחק).

}

// ---------- API ----------
function doPost(e) {
  var out;
  try {
    var req = JSON.parse(e.postData.contents);
    var AUTH_FREE = ['login', 'validateToken', 'loginByRemember', 'rememberLogin'];
    var verifiedUser = req.user || '';
    var payload = req.payload || {};
    var hasValidSession = false;
    var tk = req.token || payload.token || '';
    if (tk) {
      var sessionData = CacheService.getScriptCache().get('session_' + tk);
      if (sessionData) {
        var sd = JSON.parse(sessionData);
        payload._verified_employee_id = sd.employee_id;
        payload._verified_role = sd.role;
        verifiedUser = sd.name || sd.role;
        hasValidSession = true;
      }
    }
    if (String(req.action || '').indexOf('portal') === 0) {
      out = route(req.action, payload, 'פורטל');
    } else if (AUTH_FREE.indexOf(req.action) > -1) {
      out = route(req.action, payload, verifiedUser);
    } else if (!hasValidSession && req.password !== PASSWORD) {
      out = { ok: false, error: 'סיסמה שגויה' };
    } else out = route(req.action, payload, verifiedUser);
  } catch (err) { out = { ok: false, error: String(err) }; }
  return ContentService.createTextOutput(JSON.stringify(out)).setMimeType(ContentService.MimeType.JSON);
}

var READ_ONLY_ACTIONS = ['getAll', 'getFullTexts', 'login', 'availability', 'scan', 'reports', 'ordersReport', 'findOrderByScan', 'portalGetData', 'portalCatalog', 'validateToken', 'loginByRemember', 'rememberLogin', 'nobleCartInfo', 'nobleBoard',

  // B9: דוחות ניהול — קריאה בלבד (E3 חובות לפי גיל, E4 דשבורד חריגים)
  'b9AgingReport', 'b9ExceptionsDashboard'];

function route(action, p, user) {
  var db = loadDb();
  var lock = null;
  if (READ_ONLY_ACTIONS.indexOf(action) === -1) {   // F1: נעילה רק לפעולות שמשנות נתונים
    lock = LockService.getScriptLock();
    lock.waitLock(20000);
  }
  try {
    // הרשאות: פעולות מנהל דורשות עובד בתפקיד מנהל (אם קיימים עובדים)
    var READ_ACTIONS = ['getAll', 'getFullTexts', 'availability', 'scan', 'reports', 'ordersReport', 'login', 'clock', 'portalGetData', 'portalCreateOrder'];
    if (READ_ACTIONS.indexOf(action) === -1 && db.employees.length && user && user !== 'פורטל') {
      var actor = db.employees.find(function (x) { return x.name === user && x.active !== 'לא'; });
      if (actor && actor.can_edit === 'לא') return { ok: false, error: 'למשתמש זה הרשאת צפייה בלבד' };
    }
    if (MANAGER_ONLY.indexOf(action) > -1 && db.employees.length) {
      var emp = db.employees.find(function (x) { return x.name === user && x.active !== 'לא'; });
      if (!emp || emp.role !== 'מנהל') return { ok: false, error: 'פעולה זו מותרת למנהל בלבד' };
    }
    var r = handle(action, p, db, user);
    if (r && r.ok && READ_ONLY_ACTIONS.indexOf(action) === -1) {  // F1: audit רק לפעולות כתיבה (שמחזיקות נעילה)
      audit(db, user, action, r.auditNote || JSON.stringify(p).substring(0, 180));
      bumpDataVersion(); // B7: מסמן לכל הלקוחות המחוברים שיש נתונים חדשים למשוך
    }
    if (r) delete r.auditNote;
    return r;
  } finally { if (lock) lock.releaseLock(); }
}

function handle(action, p, db, user) {
  switch (action) {
    case 'getAll': return { ok: true, db: getAllPayload(db) }; // G1: שדות ארוכים נשלחים כתצוגה מקדימה

    // G1: שליפת הטקסט המלא של רשומות ספציפיות (רק שדות הטקסט המוגדרים — לא רשומה שלמה)
    case 'getFullTexts': {
      var ftFields = LONG_TEXT_FIELDS[p.table];
      if (!ftFields) return { ok: false, error: 'טבלה לא נתמכת בשליפת טקסטים' };
      var ftIds = (p.ids || []).map(String);
      if (!ftIds.length) return { ok: true, texts: {} };
      var texts = {};
      db[TMAP[p.table]].forEach(function (r) {
        if (ftIds.indexOf(String(r.id)) > -1) {
          var o = {};
          ftFields.forEach(function (f) { o[f] = r[f]; });
          texts[r.id] = o;
        }
      });
      return { ok: true, texts: texts };
    }

    case 'login': {
      if (!db.employees.length) return { ok: true, user: 'מנהל', role: 'מנהל' }; // אין עובדים מוגדרים — גישה מלאה
      // מצב חדש: כניסה לפי username + password
      if (p.username) {
        var empU = db.employees.find(function (x) { return x.username && String(x.username) === String(p.username) && x.active !== 'לא'; });
        if (!empU) return { ok: false, error: 'שם משתמש או סיסמה שגויים' };
        if (!empU.password_hash) return { ok: false, error: 'סיסמה לא הוגדרה לעובד זה' };
        if (sha256WithSalt(String(p.password || ''), empU.id) !== String(empU.password_hash)) return { ok: false, error: 'שם משתמש או סיסמה שגויים' };
        var token = Utilities.getUuid();
        CacheService.getScriptCache().put('session_' + token, JSON.stringify({ employee_id: empU.id, role: empU.role, name: empU.name }), 21600);
        return { ok: true, token: token, employee_id: empU.id, role: empU.role, name: empU.name, user: empU.name, views: empU.views || '', can_edit: empU.can_edit !== 'לא', is_super_admin: isSuperAdmin(empU.id) };
      }
      // מצב ישן (fallback): כניסה לפי PIN
      var emp = db.employees.find(function (x) { return String(x.pin) === String(p.pin) && x.active !== 'לא'; });
      if (!emp) return { ok: false, error: 'קוד עובד שגוי' };
      return { ok: true, user: emp.name, role: emp.role, views: emp.views || '', can_edit: emp.can_edit !== 'לא' };
    }

    case 'create': {
      if (!TABLES[p.table]) return { ok: false, error: 'טבלה לא קיימת' };
      if (p.table === 'invoices' || p.table === 'audit_log' || p.table === 'counters'
        || p.table === 'payments')   // B2: קבלה נוצרת רק דרך addPayment / b2ProcessPayment (מספור רציף + idempotency)
        return { ok: false, error: 'מסמך זה נוצר רק דרך פעולה ייעודית' };
      var row = p.row || {};
      if (['carts', 'machines', 'vehicles'].indexOf(p.table) === -1)
        row.id = row.id || genId(p.table.substring(0, 4).toUpperCase());
      var NAME_UNIQUE = { customers: 'לקוח', suppliers: 'ספק', items: 'פריט' };
      if (NAME_UNIQUE[p.table] && row.name) {
        var nm = String(row.name).trim();
        row.name = nm;
        if (db[TMAP[p.table]].some(function (x) { return String(x.name).trim() === nm && x.active !== 'לא'; }))
          return { ok: false, error: NAME_UNIQUE[p.table] + ' בשם "' + nm + '" כבר קיים במערכת' };
      }
      if (p.table === 'customers' && !row.portal_code) {
        row.portal_code = ('C' + Math.random().toString(36).substring(2, 8)).toUpperCase();
      }
      if (p.table === 'notes') {
        if (!p.row.text || !String(p.row.text).trim()) return { ok: false, error: 'תרשומת ריקה' };
        if (!p.row.entity_type || !p.row.entity_id) return { ok: false, error: 'תרשומת חייבת להיות מקושרת' };
        row.ts = nowTs(); row.user = user || '';
      }
      if (p.table === 'task_comments') {
        if (!p.row.task_id || !p.row.text) return { ok: false, error: 'הערה ריקה' };
        row.ts = nowTs(); row.user = user || '';
      }
      if (p.table === 'tasks') {
        row.status = row.status || 'פתוחה';
        row.created_by = user || ''; row.created_at = nowTs();
      }
      if (p.table === 'meetings') row.created_by = user || '';
      if (p.table === 'links') {
        if (!row.from_type || !row.from_id || !row.to_type || !row.to_id)
          return { ok: false, error: 'תיוג חייב מקור ויעד' };
        if (String(row.from_type) === String(row.to_type) && String(row.from_id) === String(row.to_id))
          return { ok: false, error: 'אי אפשר לתייג דבר לעצמו' };
        if (db.links.some(function (l) { return l.from_type === row.from_type && l.from_id === row.from_id && l.to_type === row.to_type && l.to_id === row.to_id; }))
          return { ok: false, error: 'התיוג הזה כבר קיים' };
        row.created = nowTs(); row.user = user || '';
      }
      if (p.table === 'orders') {
        row.status = row.status === 'הצעת מחיר' ? 'הצעת מחיר' : 'טיוטה';
        row.created_at = nowTs(); // NC4: חותמת זמן מלאה
        if (!row.warehouse_id) {
          var wc = db.customers.find(function (x) { return x.id === row.customer_id; });
          row.warehouse_id = (wc && wc.warehouse_id) || '';
        }
        if (!row.start_date || !row.end_date) return { ok: false, error: 'חובה שני תאריכים' };
        if (row.end_date < row.start_date) return { ok: false, error: 'תאריך ההחזרה לפני תאריך ההתחלה' };
        row.order_number = nextNumber(db, 'order', 1001);   // CR-16b: מספר הזמנה רץ
      }
      if (p.table === 'order_lines') {
        var po = db.orders.find(function (x) { return x.id === row.order_id; });
        if (!po) return { ok: false, error: 'הזמנה לא נמצאה' };
        if (po.status !== 'טיוטה' && po.status !== 'הצעת מחיר') return { ok: false, error: 'מוסיפים פריטים רק בטיוטה או בהצעת מחיר' };
        if (!(Number(row.qty) > 0)) return { ok: false, error: 'כמות חייבת להיות גדולה מאפס' };

        try {

            'שלום ' + (cAp.name || '') + ',\nההזמנה שלך אושרה.\nתאריך אספקה: ' + oo.start_date + '\nתאריך איסוף: ' + oo.end_date);

        db.deliveries.forEach(function (d) { if (d.order_id === p.order_id && d.kind === 'אספקה' && d.status === 'מתוכנן') { d.status = 'בוצע'; if (!d.note_number) d.note_number = nextNumber(db, 'delivery_note', 3001); } });
        writeTable('deliveries', db.deliveries);

        try {
          var oDv = r2.order;
          var cDv = db.customers.find(function (x) { return x.id === oDv.customer_id; });
          if (cDv && cDv.email) sendMail(cDv.email, 'הזמנתך סופקה' + (oDv.order_number ? ' — הזמנה #' + oDv.order_number : ''),
            'שלום ' + (cDv.name || '') + ',\nההזמנה שלך סופקה.\nתאריך איסוף מתוכנן: ' + oDv.end_date);
        } catch (e) { }

      if (db.invoices.some(function (v) { return v.order_id === oc.id && v.status !== 'בוטלה'; }))
        return { ok: false, error: 'קיימת חשבונית פתוחה — בטל אותה קודם' };
      oc.status = 'בוטלה'; writeTable('orders', db.orders);
      db.deliveries.forEach(function (d) { if (d.order_id === oc.id && d.status === 'מתוכנן') d.status = 'בוטל'; });

    case 'advanceLaundry': {

        if (r5.stage === 'מוכן') {

    // ---- חשבונאות ----

        var balL = nRound2(Number(ivL.total) - paidL);
        if (Number(p.amount) > balL + 0.01) return { ok: false, error: 'הסכום גבוה מהיתרה (' + balL + ' ₪)' };
        var payL = {

        };

        return { ok: true, row: payL, balance: nRound2(balL - Number(p.amount)), auditNote: 'קבלה ' + payL.receipt_number + ' · חשבונית ' + ivL.number };

      var bal = orderBalance(db, p.order_id);

      var inv2 = db.invoices.find(function (v) { return v.order_id === p.order_id && v.status !== 'בוטלה'; });

      };

      if (inv2 && orderBalance(db, p.order_id) <= 0.01) { inv2.status = 'שולמה'; writeTable('invoices', db.invoices); }
      return { ok: true, row: pay, balance: orderBalance(db, p.order_id), auditNote: 'קבלה ' + pay.receipt_number };

        var resp = UrlFetchApp.fetch('https://www.hebcal.com/hebcal?v=1&cfg=json&maj=on&min=off&mod=off&nx=off&mf=off&ss=off&i=on&year=' + y);
        var data = JSON.parse(resp.getContentText());
        (data.items || []).forEach(function (it) {
          if (it.category === 'holiday' && it.yomtov) dates.push(String(it.date).substring(0, 10));
        });
      } catch (e) { /* אין רשת — שבתות בלבד */ }
      st = { key: 'holidays_' + y, value: dates.join(',') };
      db.settings.push(st);
      writeTable('settings', db.settings);
    }
    String(st.value || '').split(',').forEach(function (d) { if (d) set[d] = 1; });
  });
  return set;
}
function isBusinessDay(iso, holSet) {
  var d = new Date(iso + 'T12:00:00');
  return d.getDay() !== 6 && !holSet[iso]; // שבת או חג — לא יום עסקים
}
function addBusinessDays(db, startIso, n) {
  var holSet = getHolidaySet(db, startIso, String(Number(startIso.substring(0, 4)) + 1) + '-01-01');
  var d = new Date(startIso + 'T12:00:00');
  var added = 0;
  while (added < n) {
    d.setDate(d.getDate() + 1);
    var iso = d.toISOString().slice(0, 10);
    if (isBusinessDay(iso, holSet)) added++;
  }
  return d.toISOString().slice(0, 10);
}

function calcHours(inT, outT) {
  var a = String(inT).split(':'), b = String(outT).split(':');
  var mins = (Number(b[0]) * 60 + Number(b[1])) - (Number(a[0]) * 60 + Number(a[1]));
  return Math.round((mins / 60) * 100) / 100;
}

// CR-14: חישוב שעות חוצה-חצות (משמרת לילה)
function calcHoursSpan(inDate, inT, outDate, outT) {
  var a = new Date(inDate + 'T' + inT + ':00');
  var b = new Date(outDate + 'T' + outT + ':00');
  return Math.round(((b - a) / 3600000) * 100) / 100;
}

var STAGES = ['התקבל', 'בכביסה', 'בייבוש', 'בגיהוץ וקיפול', 'מוכן'];
var RESERVING_STATUSES = ['מאושרת', 'סופקה'];

function totalStock(db, itemId, wh) {
  return db.stockMoves.filter(function (m) { return m.item_id === itemId && (wh === undefined || String(m.warehouse_id || '') === String(wh || '')); })
    .reduce(function (s, m) { return s + Number(m.qty); }, 0);
}
function datesOverlap(aStart, aEnd, bStart, bEnd) { return aStart <= bEnd && aEnd >= bStart; }
function reservedQty(db, itemId, start, end, excludeOrderId, wh) {
  var qty = 0;
  db.orderLines.forEach(function (l) {
    if (l.item_id !== itemId) return;
    var o = db.orders.find(function (x) { return x.id === l.order_id; });
    if (!o || o.type !== 'השכרה') return;
    if (wh !== undefined && String(o.warehouse_id || '') !== String(wh || '')) return;
    if (excludeOrderId && o.id === excludeOrderId) return;
    if (RESERVING_STATUSES.indexOf(o.status) === -1) return;
    if (datesOverlap(o.start_date, o.end_date, start, end)) qty += Number(l.qty);
  });
  return qty;
}
function inLaundryQty(db, itemId, wh) {
  return db.laundryTasks.filter(function (t) { return t.item_id === itemId && t.stage !== 'מוכן' && (wh === undefined || String(t.warehouse_id || '') === String(wh || '')); })
    .reduce(function (s, t) { return s + Number(t.qty); }, 0);
}
function availableQty(db, itemId, start, end, excludeOrderId, wh) {
  return totalStock(db, itemId, wh) - reservedQty(db, itemId, start, end, excludeOrderId, wh) - inLaundryQty(db, itemId, wh);
}
function approveOrder(db, orderId) {
  var o = db.orders.find(function (x) { return x.id === orderId; });
  if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };
  if (o.status !== 'טיוטה') return { ok: false, error: 'רק טיוטה ניתן לאשר. סטטוס נוכחי: ' + o.status };
  // B2 (D2/S2): הזמנת פורטל/חנות שטרם עברה תשלום או מסגרת אשראי — חסומה לאישור.
  // המוצא למשרד/מנהל: גביית תשלום במסך התשלום, או סימון "תשלום בהמשך" (b2MarkPayLater).
  if (['ממתין לתשלום', 'ממתין לתשלום הפרש', 'ממתין לאישור חריגה'].indexOf(String(o.payment_status || '')) > -1)
    return { ok: false, error: 'ההזמנה טרם עברה תשלום/מסגרת אשראי. גבה תשלום במסך התשלום או סמן "תשלום בהמשך" (מנהל/משרד)' };
  if (!db.orderLines.some(function (l) { return l.order_id === orderId; })) return { ok: false, error: 'אין פריטים בהזמנה' };
  var problems = [];
  if (o.type === 'השכרה') {
    db.orderLines.filter(function (l) { return l.order_id === orderId; }).forEach(function (l) {
      var useWh = db.warehouses.length > 0;
      var avail = availableQty(db, l.item_id, o.start_date, o.end_date, orderId, useWh ? (o.warehouse_id || '') : undefined);
      if (Number(l.qty) > avail) {
        var item = db.items.find(function (i) { return i.id === l.item_id; });
        var msg = (item ? item.name : l.item_id) + ': מבוקש ' + l.qty + ', פנוי ' + avail + (useWh ? ' במחסן ' + whName(db, o.warehouse_id) : '');
        if (useWh) {
          var alts = db.warehouses.filter(function (w) { return w.id !== o.warehouse_id && w.active !== 'לא'; })
            .map(function (w) { var a = availableQty(db, l.item_id, o.start_date, o.end_date, orderId, w.id); return a > 0 ? w.name + ': ' + a : null; })
            .filter(function (x) { return x; });
          if (alts.length) msg += '. פנוי במחסנים אחרים — ' + alts.join(', ') + ' (בצע העברת מלאי ואשר שוב)';
        }
        problems.push(msg);
      }
    });
  }
  if (problems.length) return { ok: false, error: 'אין מספיק מלאי פנוי — ' + problems.join(' | ') };
  o.status = 'מאושרת';
  return { ok: true, order: o };
}

  var w = db.warehouses.find(function (x) { return x.id === wid; });
  return w ? w.name : wid;
}

  if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };

function receiveReturn(db, orderId, returns, todayStr, genIdFn) {
  var o = db.orders.find(function (x) { return x.id === orderId; });
  if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };
  if (o.status !== 'סופקה') return { ok: false, error: 'רק הזמנה שסופקה ניתן להחזיר' };
  var shortages = [];
  db.orderLines.filter(function (l) { return l.order_id === orderId; }).forEach(function (l) {
    var ret = returns && returns[l.id] !== undefined ? Number(returns[l.id]) : Number(l.qty);
    if (ret < 0 || ret > Number(l.qty)) { ret = Math.max(0, Math.min(ret, Number(l.qty))); }
    l.returned_qty = ret;
    var missing = Number(l.qty) - ret;
    if (missing > 0) {
      db.stockMoves.push({ id: genIdFn('MOV'), date: todayStr, item_id: l.item_id, type: 'גריעה', qty: -missing, ref: orderId, note: 'חוסר בהחזרה' });
      var chItem = db.items.find(function (i) { return i.id === l.item_id; }) || {};
      var chPrice = Number(chItem.replacement_price) || Number(chItem.rent_price) || Number(l.unit_price) || 0;
      shortages.push({ item_id: l.item_id, missing: missing, charge: missing * chPrice });
    }
    if (ret > 0) {
      db.laundryTasks.push({
        id: genIdFn('WASH'), order_id: orderId, item_id: l.item_id, qty: ret, stage: 'התקבל', cart_id: '', machine_id: '', worker: '', in_date: todayStr, done_date: '',
        warehouse_id: o.warehouse_id || '', target_date: addBusinessDays(db, todayStr, Number((db.settings.find(function (s) { return s.key === 'laundry_days'; }) || {}).value || 2))
      });
    }
  });
  o.status = 'הוחזרה';
  o.shortage_charge = Math.round(shortages.reduce(function (s, x) { return s + x.charge; }, 0) * 100) / 100;
  return { ok: true, shortages: shortages, shortage_charge: o.shortage_charge };
}
function assignCart(db, taskId, cartId) {
  var t = db.laundryTasks.find(function (x) { return x.id === taskId; });
  var c = db.carts.find(function (x) { return x.id === cartId; });
  if (!t || !c) return { ok: false, error: 'משימה או עגלה לא נמצאו' };
  if (c.status !== 'פנויה') return { ok: false, error: 'העגלה ' + cartId + ' אינה פנויה' };
  if (t.cart_id) { releaseCart(db, t.cart_id); }
  t.cart_id = cartId;
  c.status = 'בשימוש';
  return { ok: true };
}
function releaseCart(db, cartId) {
  var c = db.carts.find(function (x) { return x.id === cartId; });
  if (c) c.status = 'פנויה';
}
function advanceLaundry(db, taskId, todayStr) {
  var t = db.laundryTasks.find(function (x) { return x.id === taskId; });
  if (!t) return { ok: false, error: 'משימה לא נמצאה' };
  var idx = STAGES.indexOf(t.stage);
  if (idx === -1 || idx === STAGES.length - 1) return { ok: false, error: 'המשימה כבר בשלב מוכן' };
  t.stage = STAGES[idx + 1];
  if (t.stage === 'מוכן') {
    t.done_date = todayStr;
    if (t.cart_id) { releaseCart(db, t.cart_id); t.cart_id = ''; }
    var o = db.orders.find(function (x) { return x.id === t.order_id; });
    if (o) {
      var open = db.laundryTasks.filter(function (x) { return x.order_id === o.id && x.stage !== 'מוכן'; });
      if (open.length === 0 && o.status === 'הוחזרה') o.status = 'הושלמה';
    }
  }
  return { ok: true, stage: t.stage };
}
function orderTotal(db, orderId) {
  var o = db.orders.find(function (x) { return x.id === orderId; });
  var lines = db.orderLines.filter(function (l) { return l.order_id === orderId; })
    .reduce(function (s, l) { return s + Number(l.qty) * Number(l.unit_price); }, 0);
  return lines + Number((o && o.delivery_fee) || 0) + Number((o && o.shortage_charge) || 0);
}

function orderBalance(db, orderId) {
  // כשקיימת חשבונית פעילה — היתרה היא מול סכום החשבונית (כולל מע"מ); אחרת מול סכום ההזמנה
  var inv = db.invoices ? db.invoices.find(function (v) { return v.order_id === orderId && v.status !== 'בוטלה'; }) : null;
  var base = inv ? Number(inv.total) : orderTotal(db, orderId);
  return base - orderPaid(db, orderId);
}
function customerBalance(db, customerId) {
  var b = db.orders.filter(function (o) { return o.customer_id === customerId && o.status !== 'בוטלה' && o.status !== 'טיוטה' && o.status !== 'הצעת מחיר'; })
    .reduce(function (s, o) { return s + orderBalance(db, o.id); }, 0);
  // NC4: חשבוניות מכבסה מרוכזות (ללא הזמנה) נכללות ביתרת הלקוח — כמו בממשק
  db.invoices.filter(function (v) { return v.customer_id === customerId && !v.order_id && v.status !== 'בוטלה'; }).forEach(function (v) {
    var paid = db.payments.filter(function (x) { return x.invoice_id === v.id; }).reduce(function (s, x) { return s + (Number(x.amount) || 0); }, 0);
    b += Math.max(0, Number(v.total) - paid);
  });
  return b;
}

function nRound2(n) { return Math.round(Number(n) * 100) / 100; }

// זיהוי העובד המבצע: קוד אישי (מצב טאבלט משותף) גובר על סשן אישי
function nobleWorker(db, p, user) {
  if (p.worker_pin !== undefined && String(p.worker_pin) !== '') {
    var byPin = db.employees.find(function (x) { return String(x.pin) === String(p.worker_pin) && x.active !== 'לא'; });
    if (!byPin) return { ok: false, error: 'קוד עובד שגוי' };
    return { ok: true, id: byPin.id, name: byPin.name };
  }
  if (p._verified_employee_id) {
    var bySess = db.employees.find(function (x) { return x.id === p._verified_employee_id; });
    if (bySess) return { ok: true, id: bySess.id, name: bySess.name };
  }
  if (db.employees.length) {
    var byName = db.employees.find(function (x) { return x.name === user && x.active !== 'לא'; });
    if (byName) return { ok: true, id: byName.id, name: byName.name };
    return { ok: false, error: 'נדרש זיהוי עובד — הזן קוד אישי' };
  }
  return { ok: true, id: '', name: user || 'מנהל' }; // אין עובדים מוגדרים — מצב הקמה
}

// פענוח קלט: ברקוד עגלה או מזהה קליטה → הקליטה הפעילה (+העגלה אם נסרקה)
function nobleResolveIntake(db, p) {
  if (p.intake_id) {
    var ik = nobleIntakeById(db, p.intake_id);
    return ik ? { intake: ik, cart: null } : { error: 'קליטה לא נמצאה' };
  }
  var cart = nobleFindCart(db, p.cart_barcode);
  if (!cart) return { error: 'עגלה לא נמצאה — בדוק את הברקוד' };
  var bind = nobleActiveBinding(db, cart.id);
  if (!bind) return { error: 'העגלה ' + cart.id + ' אינה משויכת לקליטה פעילה' };
  var intake = nobleIntakeById(db, bind.intake_id);
  if (!intake) return { error: 'הקליטה של העגלה לא נמצאה' };
  return { intake: intake, cart: cart };
}

// אירוע "התחלה" פתוח (ללא "סיום" תואם) עבור עגלה בקליטה — לפי סדר היומן
function nobleOpenStart(db, intakeId, cartId) {
  var open = null;
  nobleEventsOf(db, intakeId, cartId).forEach(function (e) {
    if (e.event_type === 'התחלה') open = e;
    else if (e.event_type === 'סיום' && open && e.stage === open.stage) open = null;
  });

}
// חישוב סיכום שקילות מהיומן: השקילה האחרונה של כל עגלה קובעת (שקילה חוזרת מחליפה)
function nobleWeighRollup(db, intakeId) {
  var lastByCart = {};
  nobleEventsOf(db, intakeId).forEach(function (e) {
    if (e.event_type === 'שקילה') lastByCart[e.cart_id] = e;
  });
  var net = 0, charge = 0, count = 0;
  Object.keys(lastByCart).forEach(function (c) {
    net += Number(lastByCart[c].net_kg) || 0;
    charge += Number(lastByCart[c].charge) || 0;
    count++;
  });
  return { net: nRound2(net), charge: nRound2(charge), count: count };
}

// --- פעולת קריאה: מידע לעגלה סרוקה (למסך הרצפה) ---
function nobleCartInfo(db, p) {
  var cart = nobleFindCart(db, p.barcode);
  if (!cart) return { ok: false, error: 'עגלה לא נמצאה — בדוק את הברקוד' };
  var bind = nobleActiveBinding(db, cart.id);
  if (!bind) return { ok: true, cart: cart, intake: null };
  var intake = nobleIntakeById(db, bind.intake_id);
  var cust = intake && intake.customer_id ? db.customers.find(function (c) { return c.id === intake.customer_id; }) : null;
  var carts = db.intakeCarts.filter(function (b) { return b.intake_id === intake.id && b.active === 'כן'; }).map(function (b) { return b.cart_id; });
  var open = nobleOpenStart(db, intake.id, cart.id);
  var roll = nobleWeighRollup(db, intake.id);
  return {
    ok: true, cart: cart, intake: intake,
    customer_name: cust ? cust.name : (intake.internal === 'כן' ? 'פנימי — MAPA' : ''),
    price_per_kg: intake.internal === 'כן' ? 0 : Number(cust && cust.price_per_kg) || 0,
    bound_carts: carts, open_stage: open ? open.stage : '', open_machine: open ? open.machine_id : '',
    weighed: roll
  };
}

// --- קליטה: פתיחת אצווה ושיוך עגלות ---
function nobleIntake(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var internal = p.internal === 'כן' || p.internal === true;
  var cust = null;
  if (!internal) {
    cust = db.customers.find(function (c) { return c.id === p.customer_id && c.active !== 'לא'; });
    if (!cust) return { ok: false, error: 'יש לבחור לקוח (או לסמן כביסה פנימית)' };
  }
  var bcs = (p.cart_barcodes || []).map(String).filter(function (s) { return s.trim() !== ''; });
  if (!bcs.length) return { ok: false, error: 'יש לסרוק לפחות עגלה אחת' };
  var carts = [], seen = {};
  for (var i = 0; i < bcs.length; i++) {
    var c = nobleFindCart(db, bcs[i]);
    if (!c) return { ok: false, error: 'עגלה לא נמצאה: ' + bcs[i] };
    if (seen[c.id]) return { ok: false, error: 'העגלה ' + c.id + ' נסרקה פעמיים' };
    seen[c.id] = 1;
    if (nobleActiveBinding(db, c.id)) return { ok: false, error: 'העגלה ' + c.id + ' כבר משויכת לקליטה פעילה' };
    if (c.status !== 'פנויה') return { ok: false, error: 'העגלה ' + c.id + ' אינה פנויה (סטטוס: ' + c.status + ')' };
    carts.push(c);
  }
  var intake = {
    id: genId('INTK'), customer_id: internal ? '' : cust.id, internal: internal ? 'כן' : '',
    status: 'התקבל', net_weight_kg: '', price_per_kg: '', total_charge: '',
    intake_ts: nowTs(), delivered_ts: '', delivery_id: '', notes: p.notes || '', created_by: w.name
  };
  db.laundryIntakes.push(intake);
  carts.forEach(function (c) {
    db.intakeCarts.push({ id: genId('IC'), intake_id: intake.id, cart_id: c.id, bind_ts: nowTs(), release_ts: '', active: 'כן' });
    c.status = 'בשימוש';
    nobleEvent(db, {
      intake_id: intake.id, customer_id: intake.customer_id, cart_id: c.id,
      stage: 'התקבל', event_type: 'קליטה', worker_id: w.id, worker_name: w.name, note: p.notes || ''
    });
  });
  writeTable('laundry_intakes', db.laundryIntakes);
  writeTable('intake_carts', db.intakeCarts);
  writeTable('carts', db.carts);
  return { ok: true, intake: intake, auditNote: 'קליטה ' + intake.id + ' · ' + (internal ? 'פנימי' : cust.name) + ' · ' + carts.length + ' עגלות' };
}

// --- סריקת עגלה + מכונה: תחילת שלב (המכונה קובעת את השלב) ---
function nobleStageStart(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var r = nobleResolveIntake(db, p); if (r.error) return { ok: false, error: r.error };
  var intake = r.intake, cart = r.cart;
  if (!cart) return { ok: false, error: 'יש לסרוק ברקוד עגלה' };
  if (intake.status === 'במשלוח' || intake.status === 'נמסר') return { ok: false, error: 'הקליטה כבר ' + intake.status };
  var mbc = String(p.machine_barcode || '').trim();
  var mach = db.machines.find(function (m) { return String(m.barcode) === mbc || String(m.id) === mbc; });
  if (!mach) return { ok: false, error: 'מכונה לא נמצאה — בדוק את הברקוד' };
  if (mach.status && mach.status !== 'פעילה') return { ok: false, error: 'המכונה ' + mach.id + ' אינה פעילה (סטטוס: ' + mach.status + ')' };
  var stage = String(mach.auto_stage || '').trim();
  if (NOBLE_MACHINE_STAGES.indexOf(stage) === -1) return { ok: false, error: 'למכונה ' + mach.id + ' לא הוגדר שלב אוטומטי — עדכן במסך עגלות ומכונות' };
  var open = nobleOpenStart(db, intake.id, cart.id);
  if (open) return { ok: false, error: 'לעגלה יש שלב פתוח (' + open.stage + ') — סיים אותו לפני תחילת שלב חדש' };
  nobleEvent(db, {
    intake_id: intake.id, customer_id: intake.customer_id, cart_id: cart.id, machine_id: mach.id,
    stage: stage, event_type: 'התחלה', worker_id: w.id, worker_name: w.name
  });
  intake.status = stage;
  writeTable('laundry_intakes', db.laundryIntakes);
  return { ok: true, stage: stage, machine_id: mach.id, intake_id: intake.id, auditNote: 'התחלת ' + stage + ' · ' + cart.id + ' → ' + mach.id };
}

// --- סיום השלב הפתוח של עגלה ---
function nobleStageEnd(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var r = nobleResolveIntake(db, p); if (r.error) return { ok: false, error: r.error };
  var intake = r.intake, cart = r.cart;
  if (!cart) return { ok: false, error: 'יש לסרוק ברקוד עגלה' };
  var open = nobleOpenStart(db, intake.id, cart.id);
  if (!open) return { ok: false, error: 'אין שלב פתוח לעגלה ' + cart.id };
  nobleEvent(db, {
    intake_id: intake.id, customer_id: intake.customer_id, cart_id: cart.id, machine_id: open.machine_id,
    stage: open.stage, event_type: 'סיום', worker_id: w.id, worker_name: w.name
  });
  return { ok: true, stage: open.stage, auditNote: 'סיום ' + open.stage + ' · ' + cart.id };
}

// --- שקילה (בשלב האריזה): ברוטו − משקל עצמי = נטו × מחיר לק"ג ---
function nobleWeigh(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var r = nobleResolveIntake(db, p); if (r.error) return { ok: false, error: r.error };
  var intake = r.intake, cart = r.cart;
  if (!cart) return { ok: false, error: 'יש לסרוק ברקוד עגלה' };
  if (intake.status === 'במשלוח' || intake.status === 'נמסר') return { ok: false, error: 'הקליטה כבר ' + intake.status };
  var gross = Number(p.gross_kg);
  if (!(gross > 0)) return { ok: false, error: 'יש להזין משקל ברוטו תקין (ק"ג)' };
  var tare = Number(cart.tare_kg) || 0;
  var net = nRound2(gross - tare);
  if (!(net > 0)) return { ok: false, error: 'משקל נטו אינו חיובי (ברוטו ' + gross + ' − משקל עצמי ' + tare + ') — בדוק את משקל העגלה' };
  var price = 0;
  if (intake.internal !== 'כן') {
    var cust = db.customers.find(function (c) { return c.id === intake.customer_id; });
    price = Number(cust && cust.price_per_kg) || 0;
    if (!(price > 0)) return { ok: false, error: 'ללקוח לא הוגדר מחיר לק"ג — עדכן בכרטיס הלקוח לפני שקילה' };
  }
  var charge = nRound2(net * price); // פנימי: מחיר 0 → חיוב 0 (נספר במשקלים, לא בהכנסות)
  nobleEvent(db, {
    intake_id: intake.id, customer_id: intake.customer_id, cart_id: cart.id,
    stage: 'באריזה', event_type: 'שקילה', worker_id: w.id, worker_name: w.name,
    gross_kg: gross, tare_kg: tare, net_kg: net, price_per_kg: price, charge: charge
  });
  var roll = nobleWeighRollup(db, intake.id); // שקילה חוזרת של אותה עגלה מחליפה את הקודמת
  intake.net_weight_kg = roll.net;
  intake.total_charge = roll.charge;
  intake.price_per_kg = price;
  if (['התקבל', 'בכביסה', 'בייבוש', 'בגיהוץ וקיפול'].indexOf(intake.status) > -1) intake.status = 'באריזה';
  writeTable('laundry_intakes', db.laundryIntakes);
  return { ok: true, net_kg: net, tare_kg: tare, charge: charge, intake_total_net: roll.net, intake_total_charge: roll.charge, auditNote: 'שקילה ' + cart.id + ' · נטו ' + net + ' ק"ג · ₪' + charge };
}

// --- סימון "מוכן למשלוח" (אחרי שקילה) ---
function nobleMarkReady(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var r = nobleResolveIntake(db, p); if (r.error) return { ok: false, error: r.error };
  var intake = r.intake;
  if (intake.status === 'מוכן') return { ok: false, error: 'הקליטה כבר מסומנת כמוכנה' };
  if (intake.status === 'במשלוח' || intake.status === 'נמסר') return { ok: false, error: 'הקליטה כבר ' + intake.status };
  var roll = nobleWeighRollup(db, intake.id);
  if (!roll.count) return { ok: false, error: 'לא ניתן לסמן מוכן — לא בוצעה שקילה לאף עגלה' };
  var openCarts = db.intakeCarts.filter(function (b) { return b.intake_id === intake.id && b.active === 'כן'; })
    .map(function (b) { return b.cart_id; })
    .filter(function (cid) { return nobleOpenStart(db, intake.id, cid); });
  if (openCarts.length) return { ok: false, error: 'יש שלבים פתוחים לעגלות: ' + openCarts.join(', ') + ' — סיים אותם קודם' };
  intake.status = 'מוכן';
  nobleEvent(db, {
    intake_id: intake.id, customer_id: intake.customer_id, cart_id: r.cart ? r.cart.id : '',
    stage: 'מוכן', event_type: 'סטטוס', worker_id: w.id, worker_name: w.name
  });
  writeTable('laundry_intakes', db.laundryIntakes);
  return { ok: true, intake: intake, auditNote: 'מוכן למשלוח · ' + intake.id };
}

// --- מסירה ללקוח: סגירת הקליטה ושחרור העגלות ---
function nobleDeliver(db, p, user) {
  var w = nobleWorker(db, p, user); if (!w.ok) return w;
  var r = nobleResolveIntake(db, p); if (r.error) return { ok: false, error: r.error };
  var intake = r.intake;
  if (intake.status !== 'במשלוח') return { ok: false, error: 'ניתן לסמן נמסר רק לקליטה במצב "במשלוח" (מצב נוכחי: ' + intake.status + ')' };
  intake.status = 'נמסר';
  intake.delivered_ts = nowTs();
  db.intakeCarts.forEach(function (b) {
    if (b.intake_id === intake.id && b.active === 'כן') {
      b.active = 'לא'; b.release_ts = nowTs();
      var c = db.carts.find(function (x) { return x.id === b.cart_id; });
      if (c) c.status = 'פנויה';
    }
  });
  if (intake.delivery_id) {
    var dlv = db.deliveries.find(function (d) { return d.id === intake.delivery_id; });
    if (dlv && dlv.status === 'מתוכנן') { dlv.status = 'בוצע'; if (!dlv.note_number) dlv.note_number = nextNumber(db, 'delivery_note', 3001); writeTable('deliveries', db.deliveries); }
  }
  nobleEvent(db, {
    intake_id: intake.id, customer_id: intake.customer_id, cart_id: '',
    stage: 'נמסר', event_type: 'מסירה', worker_id: w.id, worker_name: w.name
  });
  writeTable('laundry_intakes', db.laundryIntakes);
  writeTable('intake_carts', db.intakeCarts);
  writeTable('carts', db.carts);
  return { ok: true, intake: intake, auditNote: 'נמסר · ' + intake.id + ' · ₪' + (intake.total_charge || 0) };
}

// ============================================================
//  NC2: לוח רצפה חי · זמן משוער (ETA) · נתוני פורטל
// ============================================================
var NOBLE_WORK_STAGES = ['בכביסה', 'בייבוש', 'בגיהוץ וקיפול', 'באריזה'];
var NOBLE_ETA_DEFAULT_MIN = 60; // דקות לשלב כשאין עדיין היסטוריה ואין עקיפה ידנית

function nobleParseTs(s) { var d = new Date(String(s || '').replace(' ', 'T')); return isNaN(d.getTime()) ? null : d; }

// ממוצע משך לכל שלב מתוך זוגות התחלה→סיום ביומן. עקיפה ידנית מההגדרות גוברת.
function nobleStageAvgs(db) {
  var sums = {}, counts = {}, open = {};
  db.laundryEvents.forEach(function (e) {
    var k = e.intake_id + '|' + e.cart_id + '|' + e.stage;
    if (e.event_type === 'התחלה') open[k] = e.ts;
    else if (e.event_type === 'סיום' && open[k]) {
      var a = nobleParseTs(open[k]), b = nobleParseTs(e.ts);
      if (a && b && b > a) {
        var m = (b - a) / 60000;
        if (m >= 1 && m <= 1440) { sums[e.stage] = (sums[e.stage] || 0) + m; counts[e.stage] = (counts[e.stage] || 0) + 1; }
      }
      delete open[k];
    }
  });
  var ovRaw = (db.settings.find(function (s) { return s.key === 'noble_eta_override'; }) || {}).value || '';
  var ov = {};
  String(ovRaw).split(',').forEach(function (pair) {
    var kv = pair.split('=');
    if (kv.length === 2) { var n = Number(kv[1]); if (n > 0) ov[kv[0].trim()] = n; }
  });
  var out = {};
  NOBLE_WORK_STAGES.forEach(function (st) {
    if (ov[st]) out[st] = { min: Math.round(ov[st]), src: 'ידני' };
    else if (counts[st]) out[st] = { min: Math.round(sums[st] / counts[st]), src: 'היסטוריה (' + counts[st] + ')' };
    else out[st] = { min: NOBLE_ETA_DEFAULT_MIN, src: 'ברירת מחדל' };
  });
  return out;
}

// התקדמות עגלה בקליטה: אילו שלבים הושלמו, איזה שלב פתוח כרגע ובאיזו מכונה
function nobleCartProgress(db, intakeId, cartId) {
  var done = {}, open = null, openTs = '', mach = '';
  nobleEventsOf(db, intakeId, cartId).forEach(function (e) {
    if (e.event_type === 'התחלה') { open = e.stage; openTs = e.ts; mach = e.machine_id; }
    else if (e.event_type === 'סיום' && open === e.stage) { done[e.stage] = 1; open = null; openTs = ''; mach = ''; }
  });
  return { done: done, open: open, open_ts: openTs, machine: mach };
}

// זמן משוער (בדקות) לסיום קליטה: לכל עגלה — חצי מהשלב הפתוח + השלבים שטרם בוצעו; הקובע הוא האיטי מבין העגלות
function nobleEtaMin(db, intake, avgs) {
  if (['מוכן', 'במשלוח', 'נמסר'].indexOf(intake.status) > -1) return 0;
  var binds = db.intakeCarts.filter(function (b) { return b.intake_id === intake.id && b.active === 'כן'; });
  var eta = 0;
  binds.forEach(function (b) {
    var pr = nobleCartProgress(db, intake.id, b.cart_id), m = 0;
    NOBLE_WORK_STAGES.forEach(function (st) {
      if (pr.done[st]) return;
      m += pr.open === st ? avgs[st].min / 2 : avgs[st].min;
    });
    if (m > eta) eta = m;
  });
  return Math.round(eta);
}

// --- פעולת קריאה: תמונת מצב מלאה ללוח החי ---
function nobleBoard(db) {
  var avgs = nobleStageAvgs(db);
  function custName(id) {
    if (!id) return 'פנימי MAPA';
    var c = db.customers.find(function (x) { return x.id === id; });
    return c ? c.name : id;
  }
  var active = db.laundryIntakes.filter(function (i) { return i.status !== 'נמסר'; });
  var intakes = active.map(function (ik) {
    var binds = db.intakeCarts.filter(function (b) { return b.intake_id === ik.id && b.active === 'כן'; });
    var carts = binds.map(function (b) {
      var pr = nobleCartProgress(db, ik.id, b.cart_id);
      return { cart_id: b.cart_id, open: pr.open, open_ts: pr.open_ts, machine: pr.machine, done: Object.keys(pr.done) };
    });
    var roll = nobleWeighRollup(db, ik.id);
    return {
      id: ik.id, customer: custName(ik.customer_id), internal: ik.internal === 'כן', status: ik.status,
      intake_ts: ik.intake_ts, carts: carts, net: roll.net, charge: roll.charge, eta_min: nobleEtaMin(db, ik, avgs)
    };
  });
  var busy = {};
  active.forEach(function (ik) {
    db.intakeCarts.forEach(function (b) {
      if (b.intake_id !== ik.id || b.active !== 'כן') return;
      var pr = nobleCartProgress(db, ik.id, b.cart_id);
      if (pr.open && pr.machine) busy[pr.machine] = { intake: ik.id, customer: custName(ik.customer_id), cart: b.cart_id, since: pr.open_ts, stage: pr.open };
    });
  });
  var machines = db.machines.map(function (m) {
    return { id: m.id, type: m.type, status: m.status, auto_stage: m.auto_stage, busy: busy[m.id] || null };
  });
  var td = today(), kg = 0, kgInt = 0, evCount = 0;
  var internalSet = {};
  db.laundryIntakes.forEach(function (i) { if (i.internal === 'כן') internalSet[i.id] = 1; });
  db.laundryEvents.forEach(function (e) {
    if (String(e.ts).slice(0, 10) !== td) return;
    evCount++;
    if (e.event_type === 'שקילה') { var nk = Number(e.net_kg) || 0; kg += nk; if (internalSet[e.intake_id]) kgInt += nk; }
  });
  return {
    ok: true, intakes: intakes, machines: machines, avgs: avgs, stages: NOBLE_STAGES,
    today: {
      events: evCount, kg: nRound2(kg), kg_internal: nRound2(kgInt),
      opened: db.laundryIntakes.filter(function (i) { return String(i.intake_ts).slice(0, 10) === td; }).length,
      delivered: db.laundryIntakes.filter(function (i) { return String(i.delivered_ts).slice(0, 10) === td; }).length
    }
  };
}

// כל שקילה כבר רושמת חיוב על הקליטה (total_charge). כשהקליטה נמסרת — היא "חיוב פתוח".
// פעולה זו מרכזת את כל החיובים הפתוחים של לקוח לחשבונית אחת ומחתימה עליהם invoice_id.
function nobleCreateInvoice(db, p, user) {
  var cust = db.customers.find(function (c) { return c.id === p.customer_id; });
  if (!cust) return { ok: false, error: 'לקוח לא נמצא' };
  var open = db.laundryIntakes.filter(function (i) {
    return i.customer_id === cust.id && i.internal !== 'כן' && i.status === 'נמסר' &&
      !String(i.invoice_id || '').trim() && Number(i.total_charge) > 0;
  });
  if (!open.length) return { ok: false, error: 'אין חיובי מכבסה פתוחים ללקוח זה' };
  var sub = nRound2(open.reduce(function (s2, i) { return s2 + Number(i.total_charge); }, 0));
  var vat = nRound2(sub * VAT_RATE);
  var inv = {
    id: genId('INV'), number: nextNumber(db, 'invoice', 1001), order_id: '', customer_id: cust.id,
    date: today(), subtotal: sub, vat_rate: VAT_RATE, vat: vat, total: nRound2(sub + vat), status: 'פתוחה'
  };
  db.invoices.push(inv);
  open.forEach(function (i) { i.invoice_id = inv.id; });
  writeTable('invoices', db.invoices);
  writeTable('laundry_intakes', db.laundryIntakes);
  return { ok: true, invoice: inv, count: open.length, auditNote: 'חשבונית מכבסה ' + inv.number + ' · ' + cust.name + ' · ' + open.length + ' קליטות · ₪' + inv.total };
}

// B1: שמירת HTML → PDF ב-Drive והחזרת URL ציבורי
// הערה: Apps Script אינו תומך ב-PDF מ-HTML ישירות — שומרים HTML ב-Drive
// (בשלב ייצור ניתן להמיר ע"י Google Docs API — כאן שומרים כ-HTML עם סיומת .html)
function invoiceSaveToDrive(db, invId, htmlContent) {
  try {
    var folderName = 'MAPA-NOBLE חשבוניות';
    var folders = DriveApp.getFoldersByName(folderName);
    var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName);
    // המרה ל-PDF דרך Google Docs API (הדרך היחידה ב-Apps Script)
    var htmlBlob = Utilities.newBlob(htmlContent, 'text/html', 'invoice_' + invId);
    var docFile = Drive.Files.insert(
      { title: 'invoice_' + invId, mimeType: 'application/vnd.google-apps.document',
        parents: [{ id: folder.getId() }] },
      htmlBlob, { convert: true }
    );
    var pdfBlob = DriveApp.getFileById(docFile.id).getAs('application/pdf');
    pdfBlob.setName('invoice_' + invId + '.pdf');
    var pdfFile = folder.createFile(pdfBlob);
    pdfFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    try { DriveApp.getFileById(docFile.id).setTrashed(true); } catch(e2) {}
    return pdfFile.getUrl();
  } catch (e) {
    // Fallback: HTML אם Drive API לא מופעל
    try {
      var f2 = DriveApp.getFoldersByName('MAPA-NOBLE חשבוניות');
      var fld2 = f2.hasNext() ? f2.next() : DriveApp.createFolder('MAPA-NOBLE חשבוניות');
      var b2 = Utilities.newBlob(htmlContent, 'text/html', 'invoice_' + invId + '.html');
      var file2 = fld2.createFile(b2);
      file2.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      return file2.getUrl();
    } catch(e3) { return ''; }
  }
}

// B1: מחרוזת escape לשימוש בתוך HTML (כדי לא לייבא תלות חיצונית)
function esc2(s) {
  return String(s == null ? '' : s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}

// B1: יצירת חשבונית מורחבת — פעולה חדשה (מחליפה את createInvoice הישנה בהדרגה)
// תומכת ב: סוג מסמך, מספר הקצאה, שמירת PDF ב-Drive
function b1CreateInvoice(db, p, user) {
  var invoiceType = p.invoice_type || 'חשבונית עסקה';
  if (INVOICE_TYPES.indexOf(invoiceType) === -1) return { ok: false, error: 'סוג חשבונית לא חוקי' };

  // חישוב סכומים
  var sub, lines = [], cust = null;

  if (p.order_id) {
    // חשבונית על הזמנה
    var oi = db.orders.find(function(x) { return x.id === p.order_id; });
    if (!oi) return { ok: false, error: 'הזמנה לא נמצאה' };
    if (['הצעת מחיר', 'טיוטה', 'בוטלה'].indexOf(oi.status) > -1)
      return { ok: false, error: 'חשבונית מפיקים רק להזמנה מאושרת ומעלה' };
    if (db.invoices.some(function(v) { return v.order_id === oi.id && v.status !== 'בוטלה'; }))
      return { ok: false, error: 'כבר קיימת חשבונית פעילה להזמנה זו' };
    sub = orderTotal(db, oi.id);
    if (!(sub > 0)) return { ok: false, error: 'אין שורות בהזמנה — אין מה לחייב' };
    cust = db.customers.find(function(c) { return c.id === oi.customer_id; });
    lines = db.orderLines.filter(function(l) { return l.order_id === oi.id; }).map(function(l) {
      var it = db.items.find(function(i) { return i.id === l.item_id; }) || {};
      return { item_id: l.item_id, item_name: it.name || l.item_id, qty: l.qty, unit_price: l.unit_price };
    });
  } else {
    return { ok: false, error: 'יש לספק order_id ליצירת חשבונית. לחשבונית מכבסה השתמש ב-nobleCreateInvoice.' };
  }

  var vatRate = parseFloat((db.settings.find(function(s) { return s.key === 'vat_rate'; }) || { value: '0.18' }).value || '0.18');
  var vat = Math.round(sub * vatRate * 100) / 100;
  var draftNote = (db.settings.find(function(s) { return s.key === 'invoice_draft_note'; }) || { value: '' }).value || '';

  var inv = {
    id: genId('INV'),
    number: nextNumber(db, 'invoice', 1001),
    order_id: p.order_id || '',
    customer_id: cust ? cust.id : '',
    date: today(),
    subtotal: sub,
    vat_rate: vatRate,
    vat: vat,
    total: Math.round((sub + vat) * 100) / 100,
    status: 'פתוחה',
    // B1: שדות חדשים
    type: invoiceType,
    allocation_number: p.allocation_number || '',
    pdf_url: '',
    draft_note: draftNote
  };

  // שמירת HTML ב-Drive
  var html = buildInvoiceHtml(db, inv, lines, cust, db.settings);
  inv.pdf_url = invoiceSaveToDrive(db, inv.id, html);

  db.invoices.push(inv);
  writeTable('invoices', db.invoices);

  return {
    ok: true,
    invoice: inv,
    auditNote: invoiceType + ' ' + inv.number + ' · ' + (cust ? cust.name : '') + ' · ₪' + inv.total +
      (inv.allocation_number ? ' · הקצאה ' + inv.allocation_number : '')
  };
}

// B1: שליפת קישור HTML ל-Drive עבור חשבונית קיימת (לפורטל לקוח)

  if (!inv) return { ok: false, error: 'חשבונית לא נמצאה' };

    // יצירה מחדש אם חסר קישור
    var lines = db.orderLines.filter(function(l) { return l.order_id === inv.order_id; }).map(function(l) {
      var it = db.items.find(function(i) { return i.id === l.item_id; }) || {};
      return { item_id: l.item_id, item_name: it.name || l.item_id, qty: l.qty, unit_price: l.unit_price };

    var cust = db.customers.find(function(c) { return c.id === inv.customer_id; }) || null;

    inv.pdf_url = invoiceSaveToDrive(db, inv.id, html);

  }
  return { ok: true, url: inv.pdf_url, number: inv.number, type: inv.type };

// ============================================================
//  B2: מנוע תשלום + אשראי (D9 / D7 / D15 · הכרעות S1, S2, S5)
//  עקרונות:
//  • כל החישובים באגורות (integer) — toAg/agIls בגבולות בלבד. אין float בלוגיקה.
//  • credit_used = computed בלבד (S5): sum(יתרות הזמנות פתוחות) + חשבוניות מכבסה
//    פתוחות — זהה להגדרת customerBalance. לעולם לא שדה persistent (מונע drift).
//  • שדות amount בגיליון נשארים בש"ח — תאימות מלאה לכל המסכים הקיימים.
//  • idempotency: txn_id ייחודי נוצר בממשק עם פתיחת מסך התשלום; שליחה חוזרת
//    של אותו txn_id מחזירה את הקבלה הקיימת ואינה מחייבת שוב.
//  • קבלה אחת לטרנזקציה: קיזוז חוב על כמה הזמנות = כמה שורות payments עם אותו
//    receipt_number ואותו txn_id. מסמך הקבלה מפרט את כל ההקצאות.
// ============================================================

function toAg(n) { return Math.round(Number(n || 0) * 100); }
function agIls(ag) { return Math.round(Number(ag || 0)) / 100; }
function b2Setting(db, key) {
  var s = db.settings.find(function (x) { return x.key === key; });
  return s ? String(s.value || '') : '';
}

// יתרת הזמנה באגורות — אותה סמנטיקה כמו orderBalance: חשבונית פעילה גוברת על סכום ההזמנה
function b2OrderBalanceAg(db, orderId) {
  var inv = db.invoices.find(function (v) { return v.order_id === orderId && v.status !== 'בוטלה'; });
  var baseAg = inv ? toAg(inv.total) : toAg(orderTotal(db, orderId));
  var paidAg = db.payments.filter(function (x) { return x.order_id === orderId; })
    .reduce(function (s, x) { return s + toAg(x.amount); }, 0);
  return baseAg - paidAg;
}

// יתרת חשבונית ללא הזמנה (מכבסה מרוכזת — NC3) באגורות
function b2InvoiceBalanceAg(db, inv) {
  var paidAg = db.payments.filter(function (x) { return x.invoice_id === inv.id; })
    .reduce(function (s, x) { return s + toAg(x.amount); }, 0);
  return toAg(inv.total) - paidAg;
}

// S5: credit_used מחושב תמיד — לעולם לא נשמר בגיליון
function b2CreditUsedAg(db, customerId) {
  var used = 0;
  db.orders.filter(function (o) {
    return o.customer_id === customerId && ['בוטלה', 'טיוטה', 'הצעת מחיר'].indexOf(o.status) === -1;
  }).forEach(function (o) { used += b2OrderBalanceAg(db, o.id); });
  db.invoices.filter(function (v) { return v.customer_id === customerId && !v.order_id && v.status !== 'בוטלה'; })
    .forEach(function (v) { used += Math.max(0, b2InvoiceBalanceAg(db, v)); });
  return used;
}

// D8: limit − used = available. מסגרת צמיתה (revolving), לא מתחדשת חודשית.
function b2CreditStatusObj(db, customerId) {
  var c = db.customers.find(function (x) { return x.id === customerId; });
  var limitAg = toAg(c ? c.credit_limit : 0);
  var usedAg = b2CreditUsedAg(db, customerId);
  var pct = limitAg > 0 ? Math.round(usedAg * 100 / limitAg) : (usedAg > 0 ? 100 : 0);
  return { customer_id: customerId, limit_ag: limitAg, used_ag: usedAg, available_ag: limitAg - usedAg, pct: pct };
}

// פעולת קריאה: מצב אשראי של לקוח
function b2CreditStatus(db, p) {
  if (!p.customer_id) return { ok: false, error: 'customer_id חובה' };
  if (!db.customers.some(function (c) { return c.id === p.customer_id; })) return { ok: false, error: 'לקוח לא נמצא' };
  return { ok: true, credit: b2CreditStatusObj(db, p.customer_id) };
}

// פעולת קריאה: הקשר מלא למסך התשלום (D9) — יתרת הזמנה, חוב קודם, מצב אשראי
function b2PaymentContext(db, p) {
  var o = null;
  if (p.order_id) {
    o = db.orders.find(function (x) { return x.id === p.order_id; });
    if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };
  }
  var cid = o ? o.customer_id : p.customer_id;
  if (!cid || !db.customers.some(function (c) { return c.id === cid; })) return { ok: false, error: 'לקוח לא נמצא' };
  var credit = b2CreditStatusObj(db, cid);
  var orderBalAg = o ? Math.max(0, b2OrderBalanceAg(db, o.id)) : 0;
  // חוב קודם = החוב הכולל בניכוי יתרת ההזמנה הנוכחית (אם היא כבר נספרת בחוב)
  var counted = o && ['בוטלה', 'טיוטה', 'הצעת מחיר'].indexOf(o.status) === -1;
  var priorAg = credit.used_ag - (counted ? b2OrderBalanceAg(db, o.id) : 0);
  if (priorAg < 0) priorAg = 0;
  return {
    ok: true, customer_id: cid, credit: credit,
    order_balance_ag: orderBalAg, prior_debt_ag: priorAg,
    order_center_phone: b2Setting(db, 'order_center_phone'),
    payment_link: b2Setting(db, 'payment_link')
  };
}

// D6: מסגרת אשראי נקבעת ידנית ע"י מנהל בלבד (נאכף ב-MANAGER_ONLY)
function b2SetCreditLimit(db, p) {
  var c = db.customers.find(function (x) { return x.id === p.customer_id; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var lim = Number(p.credit_limit);
  if (!(lim >= 0) || !isFinite(lim)) return { ok: false, error: 'מסגרת אשראי חייבת להיות מספר 0 ומעלה' };
  c.credit_limit = lim;
  writeTable('customers', db.customers);
  return { ok: true, credit: b2CreditStatusObj(db, c.id), auditNote: 'מסגרת אשראי ' + c.name + ' → ₪' + lim };
}

// S2: מנהל/משרד מסמנים "תשלום בהמשך" — לאחר אישור, ההזמנה נכנסת ל-credit_used אוטומטית
// (כי credit_used מחושב מיתרות הזמנות מאושרות שלא שולמו — S5)
function b2MarkPayLater(db, p) {
  var role = String(p._verified_role || '');
  if (db.employees.length && ['מנהל', 'משרד'].indexOf(role) === -1)
    return { ok: false, error: 'סימון "תשלום בהמשך" מותר למנהל/משרד בלבד' };
  var o = db.orders.find(function (x) { return x.id === p.order_id; });
  if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };
  if (o.status === 'בוטלה') return { ok: false, error: 'הזמנה מבוטלת' };
  if (o.payment_status === 'שולם') return { ok: false, error: 'ההזמנה כבר שולמה במלואה' };
  if (o.payment_status === 'תשלום בהמשך') return { ok: true, order: o, credit: b2CreditStatusObj(db, o.customer_id) };
  o.payment_status = 'תשלום בהמשך';
  // הזמנת פורטל שטרם אושרה (הצעת מחיר) — מקדמת לטיוטה כדי שניתן לאשר אחר כך
  if (o.status === 'הצעת מחיר') o.status = 'טיוטה';
  writeTable('orders', db.orders);
  return { ok: true, order: o, credit: b2CreditStatusObj(db, o.customer_id), auditNote: 'תשלום בהמשך — הזמנה ' + (o.order_number || o.id) };
}

// לב המנוע (D9 + D15): תשלום על הזמנה ו/או קיזוז חוב קודם — קבלה אחת לכל הטרנזקציה
// payload: { customer_id, order_id?, order_amount_ag?, debt_amount_ag?, method, txn_id, note?, date? }
function b2ProcessPayment(db, p, user) {
  var cust = db.customers.find(function (c) { return c.id === p.customer_id; });
  if (!cust) return { ok: false, error: 'לקוח לא נמצא' };
  var txn = String(p.txn_id || '').trim();
  if (!txn) return { ok: false, error: 'txn_id חסר — סגור ופתח מחדש את מסך התשלום' };

  // idempotency: לחיצה כפולה / שליחה חוזרת — מחזירים את הקבלה הקיימת, לא מחייבים שוב
  var dup = db.payments.find(function (x) { return String(x.txn_id || '') === txn; });
  if (dup) return { ok: true, duplicate: true, receipt_number: dup.receipt_number, pdf_url: dup.pdf_url || '', total: '', credit: b2CreditStatusObj(db, cust.id) };

  var orderAg = Math.round(Number(p.order_amount_ag || 0));
  var debtAg = Math.round(Number(p.debt_amount_ag || 0));
  if (!(orderAg >= 0) || !(debtAg >= 0) || orderAg + debtAg <= 0) return { ok: false, error: 'סכום לתשלום חובה' };

  var o = null;
  if (orderAg > 0) {
    o = db.orders.find(function (x) { return x.id === p.order_id; });
    if (!o) return { ok: false, error: 'הזמנה לא נמצאה' };
    if (o.customer_id !== cust.id) return { ok: false, error: 'ההזמנה אינה של לקוח זה' };
    if (o.status === 'בוטלה') return { ok: false, error: 'הזמנה מבוטלת' };
    var oBal = b2OrderBalanceAg(db, o.id);
    if (orderAg > oBal) return { ok: false, error: 'הסכום גבוה מיתרת ההזמנה (₪' + agIls(Math.max(0, oBal)) + ')' };
  }

  // D15: קיזוז חוב — מותר רק במסגרת יתרת החוב הקודם (לא כולל ההזמנה הנוכחית). הקצאה FIFO — הישן קודם.
  var debts = [];
  if (debtAg > 0) {
    db.orders.filter(function (x) {
      return x.customer_id === cust.id && ['בוטלה', 'טיוטה', 'הצעת מחיר'].indexOf(x.status) === -1 && (!o || x.id !== o.id);
    }).forEach(function (x) {
      var b = b2OrderBalanceAg(db, x.id);
      if (b > 0) debts.push({ kind: 'order', id: x.id, label: 'הזמנה ' + (x.order_number || x.id), bal: b, sort: String(x.created_at || x.start_date || '') });
    });
    db.invoices.filter(function (v) { return v.customer_id === cust.id && !v.order_id && v.status !== 'בוטלה'; }).forEach(function (v) {
      var b = b2InvoiceBalanceAg(db, v);
      if (b > 0) debts.push({ kind: 'invoice', id: v.id, label: 'חשבונית ' + v.number, bal: b, sort: String(v.date || '') });
    });
    debts.sort(function (a, b) { return a.sort < b.sort ? -1 : 1; });
    var debtTotal = debts.reduce(function (s, d) { return s + d.bal; }, 0);
    if (debtAg > debtTotal) return { ok: false, error: 'קיזוז החוב גבוה מהחוב הקודם (₪' + agIls(debtTotal) + ')' };
  }

  var rcptNum = nextNumber(db, 'receipt', 5001);
  var newRows = [], docLines = [];

  if (orderAg > 0) {
    var invO = db.invoices.find(function (v) { return v.order_id === o.id && v.status !== 'בוטלה'; });
    newRows.push({
      id: genId('PAY'), order_id: o.id, invoice_id: invO ? invO.id : '', receipt_number: rcptNum,
      date: p.date || today(), amount: agIls(orderAg), method: p.method || '', note: p.note || '',
      customer_id: cust.id, type: 'תשלום הזמנה', status: 'שולם', txn_id: txn, created_by: user || '', created_at: nowTs(), pdf_url: ''
    });
    docLines.push({ desc: 'תשלום על הזמנה ' + (o.order_number || o.id), amount: agIls(orderAg) });
  }

  var remain = debtAg;
  debts.forEach(function (d) {
    if (remain <= 0) return;
    var take = Math.min(remain, d.bal); remain -= take;
    var linkedInv = '';
    if (d.kind === 'invoice') linkedInv = d.id;
    else { var iv2 = db.invoices.find(function (v) { return v.order_id === d.id && v.status !== 'בוטלה'; }); if (iv2) linkedInv = iv2.id; }
    newRows.push({
      id: genId('PAY'), order_id: d.kind === 'order' ? d.id : '', invoice_id: linkedInv,
      receipt_number: rcptNum, date: p.date || today(), amount: agIls(take), method: p.method || '', note: p.note || '',
      customer_id: cust.id, type: 'קיזוז חוב', status: 'שולם', txn_id: txn, created_by: user || '', created_at: nowTs(), pdf_url: ''
    });
    docLines.push({ desc: 'קיזוז חוב — ' + d.label, amount: agIls(take) });
  });

  newRows.forEach(function (r) { db.payments.push(r); });

  // עדכון סטטוסי חשבוניות ("שולמה") שנגעו בהן בטרנזקציה זו
  var invTouched = false;
  db.invoices.forEach(function (v) {
    if (v.status === 'בוטלה' || v.status === 'שולמה') return;
    var mine = newRows.some(function (r) { return r.invoice_id === v.id; });
    if (!mine) return;
    var bal = v.order_id ? b2OrderBalanceAg(db, v.order_id) : b2InvoiceBalanceAg(db, v);
    if (bal <= 0) { v.status = 'שולמה'; invTouched = true; }
  });
  // סטטוס תשלום של ההזמנה הנוכחית
  var ordTouched = false;
  if (o) { o.payment_status = b2OrderBalanceAg(db, o.id) <= 0 ? 'שולם' : 'שולם חלקית'; ordTouched = true; }

  // D15: הפקת קבלה מיד — HTML ל-Drive, קישור קבוע (מוצג גם בפורטל)
  var totalIls = agIls(orderAg + debtAg);
  var html = b2BuildReceiptHtml(db, { number: rcptNum, date: p.date || today(), customer: cust, method: p.method || '', lines: docLines, total: totalIls, note: p.note || '' });
  var url = invoiceSaveToDrive(db, 'RCPT-' + rcptNum, html);
  if (url) newRows.forEach(function (r) { r.pdf_url = url; });

  writeTable('payments', db.payments);
  if (invTouched) writeTable('invoices', db.invoices);
  if (ordTouched) writeTable('orders', db.orders);

  return {
    ok: true, receipt_number: rcptNum, pdf_url: url, total: totalIls,
    order_balance_ag: o ? b2OrderBalanceAg(db, o.id) : 0,
    credit: b2CreditStatusObj(db, cust.id),
    auditNote: 'קבלה ' + rcptNum + ' · ' + cust.name + ' · ₪' + totalIls + (o ? ' · הזמנה ' + (o.order_number || o.id) : '') + ' · txn ' + txn
  };
}

// שליפה/שחזור קישור קבלה (פעולת כתיבה — עשויה לשמור מחדש ב-Drive, לכן לא ב-READ_ONLY)
function b2GetReceiptLink(db, p) {
  var rows = db.payments.filter(function (x) { return String(x.receipt_number) === String(p.receipt_number); });
  if (!rows.length) return { ok: false, error: 'קבלה לא נמצאה' };
  var withUrl = rows.find(function (x) { return x.pdf_url; });
  if (withUrl) return { ok: true, url: withUrl.pdf_url, receipt_number: p.receipt_number, auditNote: 'קישור קבלה ' + p.receipt_number };
  var cust = db.customers.find(function (c) { return c.id === rows[0].customer_id; }) || null;
  if (!cust && rows[0].order_id) {
    var oo = db.orders.find(function (x) { return x.id === rows[0].order_id; });
    if (oo) cust = db.customers.find(function (c) { return c.id === oo.customer_id; }) || null;
  }
  var lines = rows.map(function (r) { return { desc: (r.type || 'תשלום') + (r.order_id ? ' — הזמנה ' + r.order_id : ''), amount: Number(r.amount || 0) }; });
  var total = nRound2(rows.reduce(function (s, r) { return s + Number(r.amount || 0); }, 0));
  var html = b2BuildReceiptHtml(db, { number: p.receipt_number, date: rows[0].date, customer: cust, method: rows[0].method || '', lines: lines, total: total, note: rows[0].note || '' });
  var url = invoiceSaveToDrive(db, 'RCPT-' + p.receipt_number, html);
  if (url) { rows.forEach(function (r) { r.pdf_url = url; }); writeTable('payments', db.payments); }
  return { ok: true, url: url, receipt_number: p.receipt_number, auditNote: 'שחזור קבלה ' + p.receipt_number };
}

// checkout פורטל — הכרעת S1 (מסגרת אשראי = תשלום תקף ל-D2) ושני מוצאי D7
// payload: { code, order_id, choice: 'credit' | 'partial' | 'approval' }
function portalB2Checkout(db, p) {
  var pc = db.customers.find(function (c) { return c.portal_code && String(c.portal_code) === String(p.code || '').toUpperCase() && c.active !== 'לא'; });
  if (!pc) return { ok: false, error: 'קוד לקוח שגוי' };
  var o = db.orders.find(function (x) { return x.id === p.order_id; });
  if (!o || o.customer_id !== pc.id) return { ok: false, error: 'הזמנה לא נמצאה' };
  if (o.status !== 'הצעת מחיר') return { ok: false, error: 'ההזמנה כבר בטיפול המשרד' };

  var totalAg = Math.max(0, b2OrderBalanceAg(db, o.id));
  var credit = b2CreditStatusObj(db, pc.id);
  var phone = b2Setting(db, 'order_center_phone');
  var payLink = b2Setting(db, 'payment_link');
  var notifyEmail = b2Setting(db, 'notify_email');

  if (p.choice === 'credit') {
    // S1: הזמנה בתוך המסגרת עוברת ללא כסף — חיוב המסגרת נחשב תשלום תקף (D2)
    if (totalAg > credit.available_ag)
      return { ok: false, over: true, gap_ag: totalAg - Math.max(0, credit.available_ag), credit: credit, order_center_phone: phone, payment_link: payLink, error: 'ההזמנה חורגת מיתרת האשראי' };
    o.status = 'טיוטה';                      // נכנסת לתור אישור המלאי של המשרד
    o.payment_status = 'באשראי מסגרת';
    writeTable('orders', db.orders);
    sendMail(notifyEmail, 'הזמנת פורטל אושרה במסגרת אשראי — #' + (o.order_number || o.id),
      'לקוח: ' + pc.name + '\nסכום (לפני מע"מ): ₪' + agIls(totalAg) + '\nיתרת מסגרת לאחר אישור: ₪' + agIls(credit.available_ag - totalAg));
    return { ok: true, mode: 'credit', credit: b2CreditStatusObj(db, pc.id), auditNote: 'checkout אשראי מסגרת · הזמנה ' + (o.order_number || o.id) + ' · ₪' + agIls(totalAg) + ' · ' + pc.name };
  }

  if (p.choice === 'partial') {
    // D7-א: תשלום ההפרש בכסף + היתרה במסגרת. הסליקה בפועל — B3; כאן נרשמת הכוונה ומוחזר קישור תשלום.
    var gap = Math.max(0, totalAg - Math.max(0, credit.available_ag));
    o.payment_status = 'ממתין לתשלום הפרש';
    writeTable('orders', db.orders);
    sendMail(notifyEmail, 'הזמנת פורטל ממתינה לתשלום הפרש — #' + (o.order_number || o.id),
      'לקוח: ' + pc.name + '\nסכום הזמנה: ₪' + agIls(totalAg) + '\nהפרש לגבייה: ₪' + agIls(gap) + '\nלאחר גבייה — לרשום במסך התשלום של ההזמנה.');
    return { ok: true, mode: 'partial', gap_ag: gap, payment_link: payLink, order_center_phone: phone, auditNote: 'checkout בקשת תשלום הפרש ₪' + agIls(gap) + ' · הזמנה ' + (o.order_number || o.id) + ' · ' + pc.name };
  }

  if (p.choice === 'approval') {
    // D7-ב: בקשת אישור חריגה — התראה למשרד (רשומת notification + מייל) + טלפון מרכז הזמנות ללקוח
    o.payment_status = 'ממתין לאישור חריגה';
    writeTable('orders', db.orders);
    db.notifications.push({
      id: genId('NTF'), type: 'בקשת חריגת אשראי', recipient: notifyEmail, channel: 'email',
      payload: JSON.stringify({ order_id: o.id, order_number: o.order_number || '', customer: pc.name, total: agIls(totalAg), available: agIls(credit.available_ag) }),
      status: 'נשלח', scheduled_for: '', sent_at: nowTs()
    });
    writeTable('notifications', db.notifications);
    sendMail(notifyEmail, 'בקשת אישור חריגת אשראי — #' + (o.order_number || o.id),
      'לקוח: ' + pc.name + '\nסכום הזמנה: ₪' + agIls(totalAg) + '\nיתרת מסגרת: ₪' + agIls(credit.available_ag) + '\nהלקוח ממתין לשיחה ממרכז ההזמנות.');
    return { ok: true, mode: 'approval', order_center_phone: phone, auditNote: 'checkout בקשת אישור חריגה · הזמנה ' + (o.order_number || o.id) + ' · ' + pc.name };
  }

  return { ok: false, error: 'בחירה לא מוכרת' };
}

//  • כרגע מצב stub: אין ספק מחובר. נקודת החיבור לספק אמיתי מסומנת ב-b3GatewayRedirectUrl.

// סטטוסים של פרופיל פקדון:
//   '' / אין רשומה  → טרם החל
//   'ממתין לאימות'   → נשלח לאימות כרטיס, טרם התקבל callback
//   'מאומת'          → token התקבל ונשמר
//   'ממתין לאישור'   → נבחרה ערבות חלופית (שיק/ערב) — ממתין לאישור מנהל (D5)
//   'מאושר'          → מנהל אישר ערבות חלופית

// שליפת לקוח פורטל לפי קוד (אותה בדיקה כמו portalGetData)
function b3PortalCustomer(db, code) {
  return db.customers.find(function (c) {
    return c.portal_code && String(c.portal_code) === String(code || '').toUpperCase() && c.active !== 'לא';
  });
}

// הפרופיל של לקוח (או null)
function b3Profile(db, customerId) {
  return db.creditProfiles.find(function (r) { return String(r.customer_id) === String(customerId); }) || null;
}

// אובייקט מצב פקדון בטוח לשליחה לממשק/פורטל — לעולם לא כולל gateway_token
function b3DepositObj(db, customerId) {
  var pr = b3Profile(db, customerId);
  if (!pr) return { exists: false, status: '', verified: false, has_token: false, guarantee_type: '', approval_pending: false, verified_at: '' };
  return {
    exists: true,
    status: pr.status || '',
    verified: !!pr.gateway_token && pr.status === 'מאומת',
    has_token: !!pr.gateway_token,
    guarantee_type: pr.guarantee_type || '',
    approval_pending: pr.approval_pending === 'כן',
    verified_at: pr.verified_at || ''
  };
}

// קריאה בלבד: מצב הפקדון של לקוח (למשרד/מנהל). מזהה לפי customer_id.
function b3DepositContext(db, p) {
  var c = db.customers.find(function (x) { return x.id === p.customer_id; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var pr = b3Profile(db, c.id);
  var obj = b3DepositObj(db, c.id);
  // למשרד נחשפים גם פרטי הערבות (לא ה-token!)
  if (pr) {
    obj.check_number = pr.check_number || '';
    obj.check_bank = pr.check_bank || '';
    obj.guarantor_name = pr.guarantor_name || '';
    obj.guarantor_phone = pr.guarantor_phone || '';
    obj.guarantor_id = pr.guarantor_id || '';
  }
  return { ok: true, customer_id: c.id, customer_name: c.name, deposit: obj };
}

// יצירה/עדכון פרופיל — מחזיר את הרשומה (בלי לכתוב לגיליון; הכתיבה אצל הקורא)
function b3EnsureProfile(db, customerId) {
  var pr = b3Profile(db, customerId);
  if (!pr) {
    pr = {
      id: genId('CRPF'), customer_id: customerId, gateway_token: '', verified_at: '',
      guarantee_type: '', check_number: '', check_bank: '',
      guarantor_name: '', guarantor_phone: '', guarantor_id: '',
      status: '', approval_pending: ''
    };
    db.creditProfiles.push(pr);
  }
  return pr;
}

// --- נקודת חיבור לספק סליקה אמיתי (S4) ---
// כשאבי יפתח חשבון ספק: שמור מפתחות ב-ScriptProperties (למשל gateway_api_key),
// קרא כאן את gateway_provider מ-settings, ובנה את כתובת ה-redirect של הספק
// (Tranzila / Cardcom / Meshulam...) עם סכום אימות ₪1 ו-callback לכתובת doGet.
// כרגע — stub: הקישור מוביל ישירות ל-doGet שמדמה חזרה מוצלחת מהספק.
function b3GatewayRedirectUrl(db, state) {
  var provider = String(b2Setting(db, 'gateway_provider') || '').trim();
  var cbUrl = ScriptApp.getService().getUrl();   // כתובת ה-Web App הפרוסה (/exec)
  if (!cbUrl) return { ok: false, error: 'לא נמצאה כתובת פריסה — ודא שה-Web App פרוס' };
  // TODO (חיבור אמיתי): switch לפי provider + credentials מ-PropertiesService.getScriptProperties()
  // return { ok: true, url: <כתובת עמוד התשלום של הספק עם state ו-callback> };
  return { ok: true, url: cbUrl + '?b3cb=1&state=' + encodeURIComponent(state) + '&stub=1', stub: true };
}

// התחלת אימות כרטיס (₪1 verification — D5): יוצר/מעדכן פרופיל במצב "ממתין לאימות"
// ומחזיר קישור redirect. state חד-פעמי נשמר ב-CacheService ל-6 שעות.
function b3StartVerification(db, customerId) {
  var c = db.customers.find(function (x) { return x.id === customerId; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var pr = b3EnsureProfile(db, c.id);
  if (pr.status === 'מאומת' && pr.gateway_token) {
    return { ok: true, already_verified: true, deposit: b3DepositObj(db, c.id) };
  }
  var state = genId('B3ST') + sha256Hex(String(Math.random()) + c.id).substring(0, 12);
  CacheService.getScriptCache().put('b3state_' + state, String(c.id), 21600);
  var rd = b3GatewayRedirectUrl(db, state);
  if (!rd.ok) return rd;
  pr.status = 'ממתין לאימות';
  writeTable('credit_profiles', db.creditProfiles);
  return {
    ok: true, redirect_url: rd.url, stub: !!rd.stub, deposit: b3DepositObj(db, c.id),
    auditNote: 'B3 התחלת אימות כרטיס · ' + c.name
  };
}

// לקוח בפורטל מבקש ערבות חלופית (שיק בטחון / ערב) — D5:
// הכרטיס עובר לאישור מנהל (approval_pending) + מופיע בדשבורד אבי (B3-3).
function b3RequestGuarantee(db, customerId, guaranteeType) {
  var c = db.customers.find(function (x) { return x.id === customerId; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var gt = String(guaranteeType || '').trim();
  if (['שיק', 'ערב'].indexOf(gt) === -1) return { ok: false, error: 'סוג ערבות לא תקין (שיק / ערב)' };
  var pr = b3EnsureProfile(db, c.id);
  pr.guarantee_type = gt;
  pr.status = 'ממתין לאישור';
  pr.approval_pending = 'כן';
  writeTable('credit_profiles', db.creditProfiles);

  db.employees.filter(function (e) { return e.role === 'מנהל' && e.active !== 'לא' && e.email; })
    .forEach(function (m) {
      sendMail(m.email, 'MAPA-NOBLE: בקשת ערבות חלופית ממתינה לאישור',
        'הלקוח ' + c.name + ' ביקש ערבות חלופית (' + gt + ').\nיש לאשר בדשבורד המערכת וליצור קשר טלפוני עם הלקוח.');
    });
  return { ok: true, deposit: b3DepositObj(db, c.id), auditNote: 'B3 בקשת ערבות חלופית (' + gt + ') · ' + c.name };
}

// Back office (משרד/מנהל): רישום פרטי שיק בטחון או ערב — D5
function b3SaveGuarantee(db, p, user) {
  var role = String(p._verified_role || '');
  if (db.employees.length && ['מנהל', 'משרד'].indexOf(role) === -1)
    return { ok: false, error: 'רישום ערבות מותר למנהל/משרד בלבד' };
  var c = db.customers.find(function (x) { return x.id === p.customer_id; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var gt = String(p.guarantee_type || '').trim();
  if (['שיק', 'ערב'].indexOf(gt) === -1) return { ok: false, error: 'סוג ערבות לא תקין (שיק / ערב)' };
  if (gt === 'שיק' && !String(p.check_number || '').trim()) return { ok: false, error: 'חסר מספר שיק' };
  if (gt === 'ערב' && !String(p.guarantor_name || '').trim()) return { ok: false, error: 'חסר שם ערב' };
  var pr = b3EnsureProfile(db, c.id);
  pr.guarantee_type = gt;
  pr.check_number = gt === 'שיק' ? String(p.check_number || '').trim() : '';
  pr.check_bank = gt === 'שיק' ? String(p.check_bank || '').trim() : '';
  pr.guarantor_name = gt === 'ערב' ? String(p.guarantor_name || '').trim() : '';
  pr.guarantor_phone = gt === 'ערב' ? String(p.guarantor_phone || '').trim() : '';
  pr.guarantor_id = gt === 'ערב' ? String(p.guarantor_id || '').trim() : '';
  pr.status = 'ממתין לאישור';
  pr.approval_pending = 'כן';
  writeTable('credit_profiles', db.creditProfiles);
  return { ok: true, deposit: b3DepositObj(db, c.id), auditNote: 'B3 רישום ערבות (' + gt + ') · ' + c.name + ' · ע"י ' + (user || '') };
}

// מנהל בלבד (נאכף ב-MANAGER_ONLY): אישור ערבות חלופית — D5
function b3ApproveGuarantee(db, p, user) {
  var c = db.customers.find(function (x) { return x.id === p.customer_id; });
  if (!c) return { ok: false, error: 'לקוח לא נמצא' };
  var pr = b3Profile(db, c.id);
  if (!pr) return { ok: false, error: 'אין פרופיל פקדון ללקוח זה' };
  if (pr.approval_pending !== 'כן') return { ok: false, error: 'אין בקשת ערבות ממתינה לאישור' };
  pr.approval_pending = '';
  pr.status = 'מאושר';
  if (!pr.verified_at) pr.verified_at = nowTs();
  writeTable('credit_profiles', db.creditProfiles);
  return { ok: true, deposit: b3DepositObj(db, c.id), auditNote: 'B3 אישור ערבות (' + (pr.guarantee_type || '') + ') · ' + c.name + ' · מאשר: ' + (user || '') };
}

// ============================================================
//  doGet — callback stub מחברת הסליקה (B3-1)
//  הספק מחזיר את הלקוח לכתובת זו אחרי אימות ₪1, עם token.
//  במצב stub: הקישור מ-b3StartVerification מגיע ישירות לכאן ומדמה הצלחה.
//  נשמר token בלבד — לעולם לא מספרי כרטיס (PCI-DSS).
// ============================================================
function doGet(e) {
  var p = (e && e.parameter) || {};
  if (String(p.b3cb) !== '1') {
    return HtmlService.createHtmlOutput('<html dir="rtl"><body style="font-family:Arial;text-align:center;padding:40px">MAPA-NOBLE API</body></html>');
  }
  var msg, okFlag = false;
  try {
    var state = String(p.state || '');
    var cache = CacheService.getScriptCache();
    var customerId = state ? cache.get('b3state_' + state) : null;
    if (!customerId) {
      msg = 'קישור האימות פג תוקף או שאינו תקין. יש להתחיל אימות מחדש מהפורטל.';
    } else {
      // token מהספק (פרמטר token/TranzilaTK וכו'). במצב stub — מייצרים token דמה.
      var token = String(p.token || '').trim();
      if (!token && String(p.stub) === '1') token = 'STUB-' + sha256Hex(state + ':' + new Date().getTime()).substring(0, 24);
      if (!token) {
        msg = 'לא התקבל אישור מחברת הסליקה. נסו שוב או פנו למשרד.';
      } else {
        var lock = LockService.getScriptLock();
        lock.waitLock(20000);
        try {
          var db = loadDb();
          var pr = b3EnsureProfile(db, customerId);
          pr.gateway_token = token;              // token בלבד — לא פרטי כרטיס
          pr.verified_at = nowTs();
          pr.status = 'מאומת';
          writeTable('credit_profiles', db.creditProfiles);
          audit(db, 'callback סליקה', 'b3VerificationCallback', 'B3 אימות כרטיס הושלם · לקוח ' + customerId + (String(p.stub) === '1' ? ' · (stub)' : ''));
        } finally { lock.releaseLock(); }
        cache.remove('b3state_' + state);
        okFlag = true;
        msg = 'אימות הכרטיס הושלם בהצלחה! ניתן לסגור חלון זה ולחזור לפורטל.';
      }
    }
  } catch (err) { msg = 'אירעה שגיאה: ' + String(err); }
  var html = '<html dir="rtl"><head><meta charset="utf-8"><meta name="viewport" content="width=device-width,initial-scale=1"></head>' +
    '<body style="font-family:Arial,sans-serif;background:#F4F6F8;display:flex;align-items:center;justify-content:center;min-height:90vh">' +
    '<div style="background:#fff;border-radius:12px;padding:36px;max-width:420px;text-align:center;box-shadow:0 2px 12px rgba(0,0,0,.08)">' +
    '<div style="font-size:44px">' + (okFlag ? '✅' : '⚠️') + '</div>' +
    '<h2 style="color:#1C2B36">' + (okFlag ? 'האימות הושלם' : 'האימות לא הושלם') + '</h2>' +
    '<p style="color:#4A5A66;font-size:15px">' + msg + '</p>' +
    (String(p.stub) === '1' ? '<p style="color:#98A6B0;font-size:12px">מצב הדגמה (stub) — טרם חובר ספק סליקה אמיתי</p>' : '') +
    '</div></body></html>';
  return HtmlService.createHtmlOutput(html).setTitle('MAPA-NOBLE — אימות כרטיס');
}

function b4Fill(tpl, vars) {
  return String(tpl || '').replace(/\{(\w+)\}/g, function (m, k) {
    return (vars[k] !== undefined && vars[k] !== null) ? String(vars[k]) : '';
  });
}

// שליחת מייל עמידה לכשל (S6: מכסה/רשת) — מחזיר true/false, לעולם לא זורק
function b4SendMailSafe(to, subject, body) {
  try {
    if (!to || String(to).indexOf('@') === -1) return false;
    MailApp.sendEmail({ to: to, subject: subject, body: body });
    return true;
  } catch (e) {
    return false;
  }
}

function b4OptedOut(c) {
  return String((c && c.notify_pref) || '').trim() === 'לא';
}

// יצירת רשומת התראה בתור, עם דדופ לפי dedupKey (נשמר בתוך payload) — מונע שכפול לאותו אירוע
function b4Queue(db, type, recipient, channel, vars, dedupKey) {
  var marker = '"dedup":"' + dedupKey + '"';
  var exists = db.notifications.some(function (n) { return n.type === type && String(n.payload || '').indexOf(marker) > -1; });
  if (exists) return null;
  var payloadObj = { dedup: dedupKey };
  Object.keys(vars || {}).forEach(function (k) { payloadObj[k] = vars[k]; });
  var row = {
    id: genId('NTF'), type: type, recipient: recipient, channel: channel || 'email',
    payload: JSON.stringify(payloadObj), status: 'ממתין', scheduled_for: today(), sent_at: ''
  };
  db.notifications.push(row);
  appendRowToTable('notifications', row);
  return row;
}

// שליחת כל ההתראות שבתור (status==='ממתין') — Phase 1: אימייל בלבד
function b4SendQueued(db) {
  var changed = false, sentCount = 0;
  db.notifications.forEach(function (n) {
    if (n.status !== 'ממתין') return;
    if (n.channel && n.channel !== 'email') return; // ערוצים נוספים (SMS/וואטסאפ) — לא ב-Phase 1
    var vars = {};
    try { vars = JSON.parse(n.payload || '{}'); } catch (e) { }
    var tpl = B4_TEMPLATES[n.type];
    if (!tpl) { n.status = 'נכשל'; changed = true; return; }
    var ok = b4SendMailSafe(n.recipient, b4Fill(tpl.subject, vars), b4Fill(tpl.body, vars));
    n.status = ok ? 'נשלח' : 'נכשל';
    n.sent_at = ok ? nowTs() : '';
    if (ok) sentCount++;
    changed = true;
  });
  if (changed) writeTable('notifications', db.notifications);
  return sentCount;
}

// תזכורת החזרה — יום (או כפי שהוגדר) לפני מועד ההחזרה, להזמנות השכרה פעילות
function b4ScanReturnReminders(db) {
  var daysBefore = Number(b2Setting(db, 'notify_return_days_before')) || 1;
  var target = Utilities.formatDate(new Date(new Date().getTime() + daysBefore * 86400000), Session.getScriptTimeZone(), 'yyyy-MM-dd');
  db.orders.filter(function (o) {
    return o.type === 'השכרה' && o.end_date === target && ['מאושרת', 'סופקה'].indexOf(o.status) > -1;
  }).forEach(function (o) {
    var c = db.customers.find(function (x) { return x.id === o.customer_id; });
    if (!c || !c.email || b4OptedOut(c)) return;
    b4Queue(db, 'return_reminder', c.email, 'email', {
      customer_name: c.name || '', order_number: o.order_number || o.id, end_date: o.end_date,
      business_name: b2Setting(db, 'business_name') || 'MAPA-NOBLE'
    }, 'RET-' + o.id + '-' + target);
  });
}

// תזכורת חוב פתוח — הזמנות שעברו X ימים ממועד ההחזרה ועדיין לא שולמו במלואן (חוזרת אחת לשבוע לכל הזמנה)
var B4_DEBT_STATUSES = ['ממתין לתשלום', 'ממתין לתשלום הפרש', 'ממתין לאישור חריגה', 'תשלום בהמשך', 'שולם חלקית', 'באשראי מסגרת'];
function b4ScanDebtReminders(db) {
  var afterDays = Number(b2Setting(db, 'notify_debt_after_days')) || 7;
  var todayMs = new Date(today()).getTime();
  db.orders.filter(function (o) {
    if (B4_DEBT_STATUSES.indexOf(o.payment_status) === -1) return false;
    if (!o.end_date) return false;
    var days = Math.floor((todayMs - new Date(o.end_date).getTime()) / 86400000);
    return days >= afterDays;
  }).forEach(function (o) {
    var c = db.customers.find(function (x) { return x.id === o.customer_id; });
    if (!c || !c.email || b4OptedOut(c)) return;
    var balAg = Math.max(0, b2OrderBalanceAg(db, o.id));
    if (balAg <= 0) return;
    var days = Math.floor((todayMs - new Date(o.end_date).getTime()) / 86400000);
    var weekBucket = Math.floor(todayMs / (7 * 86400000)); // תזכורת אחת לשבוע, לא כל יום
    b4Queue(db, 'debt_reminder', c.email, 'email', {
      customer_name: c.name || '', amount: agIls(balAg).toFixed(2), days: days,
      business_name: b2Setting(db, 'business_name') || 'MAPA-NOBLE'
    }, 'DEBT-' + o.id + '-' + weekBucket);
  });
}

// בטיחות-רשת: לקוחות שחורגים בפועל ממסגרת האשראי שלהם (מעבר להתראה בזמן-אמת שכבר קיימת ב-B2/B3) — פעם ביום למנהלים
function b4ScanCreditOverlimit(db) {
  var mgrEmails = db.employees.filter(function (e) { return e.role === 'מנהל' && e.active !== 'לא' && e.email; }).map(function (e) { return e.email; });
  if (!mgrEmails.length) return;
  db.customers.filter(function (c) { return c.active !== 'לא' && Number(c.credit_limit) > 0; }).forEach(function (c) {
    var st = b2CreditStatusObj(db, c.id);
    if (st.available_ag >= 0) return;
    mgrEmails.forEach(function (em) {
      b4Queue(db, 'credit_overlimit', em, 'email', {
        customer_name: c.name || '', limit: agIls(st.limit_ag).toFixed(2), used: agIls(st.used_ag).toFixed(2),
        over: agIls(-st.available_ag).toFixed(2)
      }, 'CRO-' + c.id + '-' + today());
    });
  });
}

// מוכן לחיבור מ-B5 (Driver-Lite): לקרוא כשמשלוח עובר לסטטוס "בדרך"
function b4NotifyOrderInTransit(db, orderId, etaText) {
  var o = db.orders.find(function (x) { return x.id === orderId; });
  if (!o) return;
  var c = db.customers.find(function (x) { return x.id === o.customer_id; });
  if (!c || !c.email || b4OptedOut(c)) return;
  b4Queue(db, 'order_in_transit', c.email, 'email', {
    customer_name: c.name || '', order_number: o.order_number || o.id, eta: etaText || 'בקרוב',
    business_name: b2Setting(db, 'business_name') || 'MAPA-NOBLE'
  }, 'TRN-' + orderId + '-' + nowTs());
}

// מוכן לחיבור מ-B5: לקרוא כשנהג מסיר כתובת ממסלול (עם פירוט חובה) — התראה מיידית להנהלה
function b4NotifyAddressRemoved(db, driverName, orderId, reason) {
  var mgrEmails = db.employees.filter(function (e) { return e.role === 'מנהל' && e.active !== 'לא' && e.email; }).map(function (e) { return e.email; });
  mgrEmails.forEach(function (em) {
    b4Queue(db, 'address_removed', em, 'email', {
      driver_name: driverName || '', order_number: orderId || '', reason: reason || ''
    }, 'ADR-' + orderId + '-' + nowTs());
  });
}

// ליבת הסריקה: מייצר התראות לתור ואז שולח את כל מה שממתין. משמש גם את הטריגר היומי וגם את ההרצה הידנית.
function b4RunScanAndSend(db) {
  var beforeCount = db.notifications.length;
  b4ScanReturnReminders(db);
  b4ScanDebtReminders(db);
  b4ScanCreditOverlimit(db);
  var queued = db.notifications.length - beforeCount;
  var sent = b4SendQueued(db);
  return { queued: queued, sent: sent };
}

// טריגר יומי (מותקן ע"י b4SetupTriggers) — רץ מחוץ ל-doPost, ולכן נועל בעצמו
function b4DailyScan() {
  var lock = LockService.getScriptLock();
  lock.waitLock(20000);
  try {
    var db = loadDb();
    var r = b4RunScanAndSend(db);
    audit(db, 'מערכת (trigger)', 'b4DailyScan', 'נוספו לתור: ' + r.queued + ' · נשלחו בהצלחה: ' + r.sent);
  } finally { lock.releaseLock(); }
}

  };

// גיבוי יומי — עותק מלא של הגיליון לתיקיית גיבוי ב-Drive, עם rotate לפי backup_keep_count.
// רץ מחוץ ל-doPost (טריגר) — אינו כותב לטבלאות עסקיות, לכן לא נדרשת LockService.
function b4DailyBackup() {
  var db = loadDb();
  var keep = Number(b2Setting(db, 'backup_keep_count')) || 30;
  var folderId = b2Setting(db, 'backup_folder_id');
  var folder = null;
  if (folderId) { try { folder = DriveApp.getFolderById(folderId); } catch (e) { folder = null; } }
  if (!folder) {
    folder = DriveApp.createFolder('MAPA-NOBLE Backups');
    var rows = db.settings;
    var s = rows.find(function (x) { return x.key === 'backup_folder_id'; });
    if (s) s.value = folder.getId(); else rows.push({ key: 'backup_folder_id', value: folder.getId() });
    writeTable('settings', rows);
  }
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var name = 'MAPA-NOBLE Backup ' + today() + ' ' + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'HHmm');
  var file = DriveApp.getFileById(ss.getId()).makeCopy(name, folder);
  // rotate: שמירת keep העותקים העדכניים ביותר בלבד
  var files = [];
  var it = folder.getFilesByType(MimeType.GOOGLE_SHEETS);
  while (it.hasNext()) files.push(it.next());
  files.sort(function (a, b) { return b.getDateCreated().getTime() - a.getDateCreated().getTime(); });
  for (var i = keep; i < files.length; i++) { try { files[i].setTrashed(true); } catch (e) { } }
  audit(db, 'מערכת (trigger)', 'b4DailyBackup', 'גיבוי הושלם: ' + file.getId() + ' · סה"כ עותקים שמורים: ' + Math.min(files.length, keep));
}

// הרצה חד-פעמית מהעורך (Apps Script) — מתקינה 2 טריגרים יומיים. בטוח להריץ שוב, לא יוצר כפילויות.
function b4SetupTriggers() {
  var existing = ScriptApp.getProjectTriggers().map(function (t) { return t.getHandlerFunction(); });
  if (existing.indexOf('b4DailyScan') === -1) ScriptApp.newTrigger('b4DailyScan').timeBased().everyDays(1).atHour(8).create();
  if (existing.indexOf('b4DailyBackup') === -1) ScriptApp.newTrigger('b4DailyBackup').timeBased().everyDays(1).atHour(3).create();
  return 'טריגרים מותקנים: ' + ScriptApp.getProjectTriggers().map(function (t) { return t.getHandlerFunction(); }).join(', ');
}

// הרשאה: מנהל/משרד תמיד יכולים; נהג רק על משלוח ששובץ לשמו
function b5CanTouch(db, dlv, user) {
  if (!dlv) return false;
  if (!db.employees.length) return true; // אין עובדים מוגדרים — גישה פתוחה (כמו שאר המערכת)
  var emp = db.employees.find(function (e) { return e.name === user && e.active !== 'לא'; });
  if (emp && ['מנהל', 'משרד'].indexOf(emp.role) > -1) return true;
  return dlv.driver && dlv.driver === user;
}

// משלוחי "היום" הפתוחים של נהג נתון (למסלול ולחישוב מיקום ב-ETA) — ממוינים לפי route_order ואז id
var B5_OPEN_STATUSES = ['מתוכנן', 'בדרך'];
function b5DriverOpenDeliveries(db, driverName) {
  return db.deliveries.filter(function (d) {
    return d.driver === driverName && B5_OPEN_STATUSES.indexOf(d.status) > -1;
  }).sort(function (a, b) {
    var ra = Number(a.route_order) || 999999, rb = Number(b.route_order) || 999999;
    if (ra !== rb) return ra - rb;
    return String(a.id) < String(b.id) ? -1 : 1;
  });
}

// קביעת סדר מסלול — p: {ids:[delivery_id,...]} לפי הסדר הרצוי. כל המזהים חייבים להשתייך לאותו נהג.
function b5SetRoute(db, p, user) {
  var ids = (p.ids || []).map(String);
  if (!ids.length) return { ok: false, error: 'לא התקבל מסלול' };
  var recs = ids.map(function (id) { return db.deliveries.find(function (x) { return String(x.id) === id; }); });
  if (recs.some(function (r) { return !r; })) return { ok: false, error: 'משלוח לא נמצא ברשימה' };
  var driverName = recs[0].driver;
  if (recs.some(function (r) { return r.driver !== driverName; })) return { ok: false, error: 'ניתן לסדר מסלול של נהג אחד בלבד בכל פעם' };
  if (!b5CanTouch(db, recs[0], user)) return { ok: false, error: 'אין הרשאה לסדר מסלול זה' };
  recs.forEach(function (r, i) { r.route_order = i + 1; });
  writeTable('deliveries', db.deliveries);
  return { ok: true, auditNote: 'B5 סדר מסלול עודכן · נהג ' + (driverName || '') + ' · ' + ids.length + ' עצירות' };
}

// סימון משלוח כ"בדרך" — מחשב eta_text לפי מיקום בין המשלוחים הפתוחים של אותו נהג, ומודיע ללקוח (אם יש הזמנה מקושרת)
function b5MarkInTransit(db, p, user) {
  var dlv = db.deliveries.find(function (x) { return String(x.id) === String(p.delivery_id); });
  if (!dlv) return { ok: false, error: 'משלוח לא נמצא' };
  if (!b5CanTouch(db, dlv, user)) return { ok: false, error: 'אין הרשאה למשלוח זה' };
  if (B5_OPEN_STATUSES.indexOf(dlv.status) === -1) return { ok: false, error: 'ניתן לסמן "בדרך" רק משלוח פתוח (מתוכנן/בדרך)' };
  dlv.status = 'בדרך';
  var route = b5DriverOpenDeliveries(db, dlv.driver);
  var pos = route.findIndex(function (r) { return String(r.id) === String(dlv.id); }) + 1;
  var total = route.length;
  var etaText = (total <= 1) ? 'יצא לדרך' : ('עצירה ' + pos + ' מתוך ' + total);
  dlv.eta_text = etaText;
  writeTable('deliveries', db.deliveries);
  if (dlv.order_id) b4NotifyOrderInTransit(db, dlv.order_id, etaText);
  return { ok: true, delivery: dlv, auditNote: 'B5 משלוח בדרך · ' + dlv.id + ' · ' + etaText };
}

// הסרת כתובת מהמסלול — חובת פירוט סיבה, מתעד ומיידע את ההנהלה (לא מוחק את המשלוח — רק יוצא מהמסלול הפתוח)
function b5RemoveFromRoute(db, p, user) {
  var dlv = db.deliveries.find(function (x) { return String(x.id) === String(p.delivery_id); });
  if (!dlv) return { ok: false, error: 'משלוח לא נמצא' };
  if (!b5CanTouch(db, dlv, user)) return { ok: false, error: 'אין הרשאה למשלוח זה' };
  if (B5_OPEN_STATUSES.indexOf(dlv.status) === -1) return { ok: false, error: 'ניתן להסיר רק משלוח פתוח (מתוכנן/בדרך)' };
  var reason = String(p.reason || '').trim();
  if (reason.length < 3) return { ok: false, error: 'יש לפרט את סיבת ההסרה (לפחות 3 תווים)' };
  dlv.status = 'הוסר ממסלול';
  dlv.note = (dlv.note ? dlv.note + ' · ' : '') + 'הוסר מהמסלול ע"י ' + (dlv.driver || user || '') + ': ' + reason;
  dlv.eta_text = '';
  writeTable('deliveries', db.deliveries);
  var ord = dlv.order_id ? db.orders.find(function (o) { return o.id === dlv.order_id; }) : null;
  b4NotifyAddressRemoved(db, dlv.driver || user, (ord && ord.order_number) || dlv.order_id || dlv.id, reason);
  return { ok: true, delivery: dlv, auditNote: 'B5 הוסר ממסלול · ' + dlv.id + ' · סיבה: ' + reason };
}

// תיעוד שיחת נהג ללקוח — ללא חשיפת מספר טלפון בממשק (כפתור חיוג משתמש ב-tel: בלבד)
var B5_CALL_OUTCOMES = ['תיאום משלוח', 'לקוח לא נמצא', 'אחר'];
function b5LogCall(db, p, user) {
  var dlv = db.deliveries.find(function (x) { return String(x.id) === String(p.delivery_id); });
  if (!dlv) return { ok: false, error: 'משלוח לא נמצא' };
  if (!b5CanTouch(db, dlv, user)) return { ok: false, error: 'אין הרשאה למשלוח זה' };
  var outcome = String(p.outcome || '').trim();
  if (B5_CALL_OUTCOMES.indexOf(outcome) === -1) return { ok: false, error: 'תוצאת שיחה לא תקינה' };
  var note = String(p.note || '').trim();
  if (outcome === 'אחר' && !note) return { ok: false, error: 'יש לפרט כשבוחרים "אחר"' };
  var emp = db.employees.find(function (e) { return e.name === user; });
  var row = {
    id: genId('CALL'), delivery_id: dlv.id, driver_id: emp ? emp.id : '', outcome: outcome, note: note, called_at: nowTs()
  };
  db.callLogs.push(row);
  appendRowToTable('call_logs', row);
  return { ok: true, call: row, auditNote: 'B5 תיעוד שיחה · משלוח ' + dlv.id + ' · ' + outcome };
}

// ============================================================
//  B9: דוחות ניהול (E3 + E4) — קריאה בלבד, בלי שינוי לוגיקה קיימת
//  E3: דוח חובות לפי גיל — שוטף (0-30) / 30+ (31-60) / 60+ (61-90) / 90+
//      תאריך הבסיס לכל יתרה: תאריך החשבונית אם קיימת, אחרת תאריך ההחזרה
//      (end_date) של ההזמנה, אחרת תאריך היצירה. חישוב באגורות (S5/S1-B2).
//  E4: דשבורד חריגים — מרכז חמישה סוגי פריטים שדורשים תשומת לב מנהל,
//      נבנה כולו מנתונים/פונקציות קיימות (B2/B3/B4/B5) — קריאה בלבד.
// ============================================================

function b9AgingBucket(days) {
  if (days <= 30) return 'current_ag';
  if (days <= 60) return 'd30_ag';
  if (days <= 90) return 'd60_ag';
  return 'd90_ag';
}

function b9AgingReport(db) {
  var todayMs = new Date(today()).getTime();
  var byCustomer = {};

  function addAmt(custId, dateStr, ag) {
    if (!custId || !(ag > 0)) return;
    if (!byCustomer[custId]) byCustomer[custId] = { current_ag: 0, d30_ag: 0, d60_ag: 0, d90_ag: 0, total_ag: 0 };
    var days = dateStr ? Math.floor((todayMs - new Date(dateStr).getTime()) / 86400000) : 0;
    if (days < 0) days = 0;
    byCustomer[custId][b9AgingBucket(days)] += ag;
    byCustomer[custId].total_ag += ag;
  }

  // הזמנות פעילות (לא בוטלה/טיוטה/הצעת מחיר) — אותה הגדרה כמו b2CreditUsedAg
  db.orders.filter(function (o) {
    return ['בוטלה', 'טיוטה', 'הצעת מחיר'].indexOf(o.status) === -1;
  }).forEach(function (o) {
    var bal = b2OrderBalanceAg(db, o.id);
    if (bal > 0) {
      var inv = db.invoices.find(function (v) { return v.order_id === o.id && v.status !== 'בוטלה'; });
      var dateStr = inv ? inv.date : (o.end_date || o.start_date || o.created_at);
      addAmt(o.customer_id, dateStr, bal);
    }
  });

  // חשבוניות מכבסה מרוכזות ללא הזמנה (NC4)
  db.invoices.filter(function (v) { return !v.order_id && v.status !== 'בוטלה'; }).forEach(function (v) {
    var bal = b2InvoiceBalanceAg(db, v);
    if (bal > 0) addAmt(v.customer_id, v.date, bal);
  });

  var rows = Object.keys(byCustomer).map(function (cid) {
    var c = db.customers.find(function (x) { return x.id === cid; });
    var b = byCustomer[cid];
    return {
      customer_id: cid, customer: c ? c.name : cid,
      current_ag: b.current_ag, d30_ag: b.d30_ag, d60_ag: b.d60_ag, d90_ag: b.d90_ag, total_ag: b.total_ag
    };
  }).filter(function (r) { return r.total_ag > 0; })
    .sort(function (a, b) { return b.total_ag - a.total_ag; });

  var totals = rows.reduce(function (s, r) {
    s.current_ag += r.current_ag; s.d30_ag += r.d30_ag; s.d60_ag += r.d60_ag; s.d90_ag += r.d90_ag; s.total_ag += r.total_ag;
    return s;
  }, { current_ag: 0, d30_ag: 0, d60_ag: 0, d90_ag: 0, total_ag: 0 });

  return { ok: true, rows: rows, totals: totals };
}

// דשבורד חריגים — חמישה סוגים, כל אחד בנוי מלוגיקה/נתונים קיימים בלבד
function b9ExceptionsDashboard(db) {
  var todayMs = new Date(today()).getTime();
  function daysOverdue(dateStr) {
    if (!dateStr) return 0;
    return Math.floor((todayMs - new Date(dateStr).getTime()) / 86400000);
  }

  // 1) לקוחות מעל 90% ניצול אשראי (D8/S5 — אותו חישוב כמו B2/B4)
  var creditOver90 = db.customers.filter(function (c) {
    return c.active !== 'לא' && Number(c.credit_limit) > 0;
  }).map(function (c) {
    var st = b2CreditStatusObj(db, c.id);
    return { customer_id: c.id, customer: c.name, pct: st.pct, used_ag: st.used_ag, limit_ag: st.limit_ag };
  }).filter(function (x) { return x.pct >= 90; })
    .sort(function (a, b) { return b.pct - a.pct; });

  // 2) החזרות שעברו את המועד ועדיין לא הוחזרו/הושלמו
  var overdueReturns = db.orders.filter(function (o) {
    return o.type === 'השכרה' && o.end_date && ['מאושרת', 'סופקה'].indexOf(o.status) > -1 && daysOverdue(o.end_date) > 0;
  }).map(function (o) {
    var c = db.customers.find(function (x) { return x.id === o.customer_id; });
    return { order_id: o.id, order_number: o.order_number || o.id, customer: c ? c.name : '', end_date: o.end_date, days: daysOverdue(o.end_date) };
  }).sort(function (a, b) { return b.days - a.days; });

  // 3) משלוחים תקועים — סטטוס פתוח (מתוכנן/בדרך) אחרי תאריך היעד
  var stuckDeliveries = db.deliveries.filter(function (d) {
    return B5_OPEN_STATUSES.indexOf(d.status) > -1 && d.date && daysOverdue(d.date) > 0;
  }).map(function (d) {
    var o = d.order_id ? db.orders.find(function (x) { return x.id === d.order_id; }) : null;
    var c = o ? db.customers.find(function (x) { return x.id === o.customer_id; }) : null;
    return { delivery_id: d.id, order_id: d.order_id || '', kind: d.kind || '', customer: c ? c.name : '', date: d.date, days: daysOverdue(d.date), status: d.status, driver: d.driver || '' };
  }).sort(function (a, b) { return b.days - a.days; });

  // 4) אישורי ערבות חלופית ממתינים למנהל (D5/B3)
  var pendingGuarantees = db.creditProfiles.filter(function (pr) {
    return pr.approval_pending === 'כן';
  }).map(function (pr) {
    var c = db.customers.find(function (x) { return x.id === pr.customer_id; });
    return { customer_id: pr.customer_id, customer: c ? c.name : '', guarantee_type: pr.guarantee_type || '' };
  });

  // 5) חשבוניות טיוטה — ממתינות לאישור תבנית מול רואה חשבון (S7)
  var draftInvoices = db.invoices.filter(function (v) {
    return v.status !== 'בוטלה' && v.draft_note && String(v.draft_note).trim();
  }).map(function (v) {
    var c = db.customers.find(function (x) { return x.id === v.customer_id; });
    return { invoice_id: v.id, number: v.number, customer: c ? c.name : '', date: v.date, total: v.total };
  }).sort(function (a, b) { return String(b.date) < String(a.date) ? -1 : 1; });

  return {
    ok: true,
    credit_over90: creditOver90,
    overdue_returns: overdueReturns,
    stuck_deliveries: stuckDeliveries,
    pending_guarantees: pendingGuarantees,
    draft_invoices: draftInvoices,
    counts: {
      credit_over90: creditOver90.length,
      overdue_returns: overdueReturns.length,
      stuck_deliveries: stuckDeliveries.length,
      pending_guarantees: pendingGuarantees.length,
      draft_invoices: draftInvoices.length
    }
  };
}
